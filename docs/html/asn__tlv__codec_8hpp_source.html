129inline bool readLength(const uint8_t* buffer, std::size_t size, std::size_t offset,
130 std::size_t& length, std::size_t& nextOffset, std::string* error)
131{
132 if (offset >= size)
133 {
134 if (error)
135 *error = "Invalid BER length offset";
136 return false;
137 }
138
139 uint8_t first = buffer[offset];
140 if ((first & 0x80U) == 0U)
141 {
142 length = first;
143 nextOffset = offset + 1U;
144 return true;
145 }
146
147 uint8_t count = first & 0x7FU;
148 if (count == 0U || offset + count >= size)
149 {
150 if (error)
151 *error = "Invalid BER length encoding";
152 return false;
153 }
154
155 std::size_t value = 0U;
156 for (uint8_t i = 0U; i < count; ++i)
157 {
158 value = (value << 8U) | buffer[offset + 1U + i];
159 }
160
161 length = value;
162 nextOffset = offset + 1U + count;
163 return true;
164}

180inline bool readTlv(const uint8_t* buffer, std::size_t size, std::size_t offset,
181 TlvInfo& info, std::string* error)
182{
183 if (offset >= size)
184 {
185 if (error)
186 *error = "Invalid TLV offset";
187 return false;
188 }
189
190 info.tag = buffer[offset];
191 info.constructed = (info.tag & 0x20U) != 0U;
192
193 std::size_t length = 0U;
194 std::size_t nextOffset = 0U;
195 if (!readLength(buffer, size, offset + 1U, length, nextOffset, error))
196 return false;
197
198 if (nextOffset + length > size)
199 {
200 if (error)
201 *error = "TLV length exceeds buffer";
202 return false;
203 }
204
205 info.valueStart = nextOffset;
206 info.length = length;
207 info.nextOffset = nextOffset + length;
208 return true;
209}

219inline void berAppendLength(std::vector<uint8_t>& out, std::size_t len)
220{
221 if (len < 128U)
222 {
223 out.push_back(static_cast<uint8_t>(len));
224 return;
225 }
226
227 std::size_t tmp = len;
228 uint8_t buf[8] = {0};
229 std::size_t count = 0U;
230
231 while (tmp > 0U && count < sizeof(buf))
232 {
233 buf[count++] = static_cast<uint8_t>(tmp & 0xFFU);
234 tmp >>= 8U;
235 }
236
237 out.push_back(static_cast<uint8_t>(0x80U | count));
238 while (count > 0U)
239 {
240 out.push_back(buf[count - 1U]);
241 --count;
242 }
243}

299inline std::vector<uint8_t> encodeSerRecordsToTlv(const std::vector<SERRecord>& records)
300{
301 std::vector<uint8_t> content;
302 content.reserve(records.size() * 64U);
303
304 for (const auto& rec : records)
305 {
306 std::vector<uint8_t> recordValue;
307 berAppendString(recordValue, 0x80U, rec.record_id);
308 berAppendString(recordValue, 0x81U, rec.timestamp);
309 berAppendString(recordValue, 0x82U, rec.status);
310 berAppendString(recordValue, 0x83U, rec.description);
311 berAppendString(recordValue, 0x84U, rec.relay_id);
312 berAppendString(recordValue, 0x85U, rec.relay_name);
313
314 std::vector<uint8_t> recordTlv;
315 berAppendTlv(recordTlv, 0x30U, recordValue.data(), recordValue.size());
316 content.insert(content.end(), recordTlv.begin(), recordTlv.end());
317 }
318
319 std::vector<uint8_t> payload;
320 berAppendTlv(payload, 0x61U, content.data(), content.size());
321 return payload;
322}

340inline bool decodeSerRecordsFromTlv(const uint8_t* data, std::size_t size,
341 std::vector<SERRecord>& out, std::string* error = nullptr)
342{
343 out.clear();
344 if (data == nullptr || size == 0U)
345 {
346 if (error)
347 *error = "Empty TLV buffer";
348 return false;
349 }
350
351 std::size_t offset = 0U;
352 TlvInfo top;
353 if (!readTlv(data, size, offset, top, error))
354 return false;
355
356 if (top.tag != 0x61U)
357 {
358 if (error)
359 *error = "Unexpected TLV tag";
360 return false;
361 }
362
363 offset = top.valueStart;
364 std::size_t end = top.valueStart + top.length;
365
366 while (offset < end)
367 {
368 TlvInfo record;
369 if (!readTlv(data, size, offset, record, error))
370 return false;
371
372 if (record.tag != 0x30U)
373 {
374 offset = record.nextOffset;
375 continue;
376 }
377
378 SERRecord rec;
379 std::size_t fieldOffset = record.valueStart;
380 std::size_t recordEnd = record.valueStart + record.length;
381
382 while (fieldOffset < recordEnd)
383 {
384 TlvInfo field;
385 if (!readTlv(data, size, fieldOffset, field, error))
386 return false;
387
388 std::string value(reinterpret_cast<const char*>(data + field.valueStart), field.length);
389
390 if (field.tag == 0x80U)
391 rec.record_id = value;
392 else if (field.tag == 0x81U)
393 rec.timestamp = value;
394 else if (field.tag == 0x82U)
395 rec.status = value;
396 else if (field.tag == 0x83U)
397 rec.description = value;
398 else if (field.tag == 0x84U)
399 rec.relay_id = value;
400 else if (field.tag == 0x85U)
401 rec.relay_name = value;
402
403 fieldOffset = field.nextOffset;
404 }
405
406 out.push_back(rec);
407 offset = record.nextOffset;
408 }
409
410 return true;
411}
