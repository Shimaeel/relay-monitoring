117 bool startRelay(const std::string& relayId)
118 {
119 std::lock_guard<std::mutex> lock(mutex_);
120
121 // Already running?
122 if (active_.count(relayId))
123 {
124 std::cout << "[RelayMgr] Relay " << relayId << " already active\n";
125 return true;
126 }
127
128 // Find config
129 const RelayConfig* cfg = nullptr;
130 for (const auto& c : configs_)
131 {
132 if (c.id == relayId)
133 {
134 cfg = &c;
135 break;
136 }
137 }
138
139 if (!cfg)
140 {
141 std::cerr << "[RelayMgr] Unknown relay ID: " << relayId << "\n";
142 return false;
143 }
144
145 // Create and start pipeline
146 auto pipeline = std::make_unique<RelayPipeline>(
147 *cfg, db_, wsServer_, shmRing_, app_running_);
148
149 if (!pipeline->start())
150 {
151 std::cerr << "[RelayMgr] Failed to start pipeline for relay " << relayId << "\n";
152 return false;
153 }
154
155 std::cout << "[RelayMgr] Started relay " << relayId
156 << " (" << cfg->name << ")\n";
157 active_[relayId] = std::move(pipeline);
158 return true;
159 }
160
169 bool stopRelay(const std::string& relayId)
170 {
171 std::lock_guard<std::mutex> lock(mutex_);
172
173 auto it = active_.find(relayId);
174 if (it == active_.end())
175 {
176 std::cout << "[RelayMgr] Relay " << relayId << " is not active\n";
177 return false;
178 }
179
180 it->second->stop();
181 active_.erase(it);
182 std::cout << "[RelayMgr] Stopped relay " << relayId << "\n";
183 return true;
184 }
