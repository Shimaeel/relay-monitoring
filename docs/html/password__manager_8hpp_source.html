91inline bool decodeASN1Password(const uint8_t* data, std::size_t size,
92 std::string& password, std::string* error = nullptr)
93{
94 if (data == nullptr || size == 0U)
95 {
96 if (error) *error = "Empty PASSWORD buffer";
97 return false;
98 }
99
100 asn_tlv::TlvInfo info;
101 if (!asn_tlv::readTlv(data, size, 0U, info, error))
102 return false;
103
104 if (info.tag != ASN1_PASSWORD_TAG)
105 {
106 if (error)
107 *error = "Unexpected tag " + std::to_string(info.tag)
108 + " (expected " + std::to_string(ASN1_PASSWORD_TAG) + ")";
109 return false;
110 }
111
112 password.assign(reinterpret_cast<const char*>(data + info.valueStart), info.length);
113 return true;
114}

222 // ── Input validation ────────────────────────────────────────────

239 // ── Relay communication ─────────────────────────────────────────
240 std::string cmd = "PAS " + level + " " + value;
241 auto cmdResult = relay_.sendRelayCommand(cmd);
242
243 if (cmdResult.success)
244 {
245 result.success = true;
246 std::cout << "[PasswordMgr] Password change successful for "
247 << level << "\n";
248 }
249 else
250 {
251 result.error = cmdResult.error.empty()
252 ? "Password change failed - no ACK from relay"
253 : cmdResult.error;
254 std::cout << "[PasswordMgr] Password change failed for "
255 << level << "\n";
256 }
257
258 return result;
259 }

313 static std::string extractJsonField(const std::string& json, const std::string& fieldName)
314 {
315 const std::string key = "\"" + fieldName + "\"";
316 auto pos = json.find(key);
317 if (pos == std::string::npos) return {};
318 pos = json.find(':', pos + key.size());
319 if (pos == std::string::npos) return {};
320 pos = json.find('"', pos + 1);
321 if (pos == std::string::npos) return {};
322 auto end = json.find('"', pos + 1);
323 if (end == std::string::npos) return {};
324 return json.substr(pos + 1, end - pos - 1);
325 }

327 static std::string escapeJson(const std::string& s)
328 {
329 std::string out;
330 out.reserve(s.size());
331 for (char c : s)
332 {
333 if (c == '"') out += "\\\"";
334 else if (c == '\\') out += "\\\\";
335 else if (c == '\n') out += "\\n";
336 else if (c == '\r') out += "\\r";
337 else out += c;
338 }
339 return out;
340 }
