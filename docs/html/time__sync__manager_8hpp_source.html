115 static std::pair<std::string, int64_t> compareTime(const std::string& relayTime,
116 const std::string& pcTime)
117 {
118 auto relayTp = parseToTimePoint(relayTime);
119 auto pcTp = parseToTimePoint(pcTime);
120
121 int64_t diffSec = std::abs(
122 std::chrono::duration_cast<std::chrono::seconds>(relayTp - pcTp).count());
123
124 std::string status = (diffSec <= 1) ? "in_sync" : "out_of_sync";
125 return {status, diffSec};
126 }

224 static std::chrono::system_clock::time_point parseToTimePoint(const std::string& dt)
225 {
226 std::tm tm{};
227 std::istringstream iss(dt);
228 iss >> std::get_time(&tm, "%Y-%m-%d %H:%M:%S");
229 if (iss.fail())
230 {
231 // Try alternate format DD/MM/YY HH:MM:SS (relay style)
232 std::istringstream iss2(dt);
233 iss2 >> std::get_time(&tm, "%m/%d/%y %H:%M:%S");
234 if (iss2.fail())
235 {
236 return std::chrono::system_clock::now(); // fallback
237 }
238 }
239 std::time_t t = std::mktime(&tm);
240 return std::chrono::system_clock::from_time_t(t);
241 }

248 static std::string escapeJson(const std::string& s)
249 {
250 std::string out;
251 out.reserve(s.size());
252 for (char c : s)
253 {
254 if (c == '"') out += "\\\"";
255 else if (c == '\\') out += "\\\\";
256 else if (c == '\n') out += "\\n";
257 else if (c == '\r') out += "\\r";
258 else out += c;
259 }
260 return out;
261 }
