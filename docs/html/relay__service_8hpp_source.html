84inline bool decodeASN1Date(const uint8_t* data, std::size_t size,
85 std::string& datetime, std::string* error = nullptr)
86{
87 if (data == nullptr || size == 0U)
88 {
89 if (error) *error = "Empty DATE buffer";
90 return false;
91 }
92
93 asn_tlv::TlvInfo info;
94 if (!asn_tlv::readTlv(data, size, 0U, info, error))
95 return false;
96
97 if (info.tag != ASN1_DATE_TAG)
98 {
99 if (error)
100 *error = "Unexpected tag " + std::to_string(info.tag)
101 + " (expected " + std::to_string(ASN1_DATE_TAG) + ")";
102 return false;
103 }
104
105 datetime.assign(reinterpret_cast<const char*>(data + info.valueStart), info.length);
106 return true;
107}

162 DateResult readRelayTime()
163 {
164 std::lock_guard<std::mutex> lock(mutex_);
165 DateResult result;
166
167 if (!client_.isConnected())
168 {
169 result.error = "Relay not connected";
170 return result;
171 }
172
173 std::string rawResponse;
174 bool ok = client_.SendCmdReceiveData("DATE", rawResponse);
175 if (!ok || rawResponse.empty())
176 {
177 result.error = "DATE command failed or empty response";
178 return result;
179 }
180
181 // The relay may respond with plain text datetime "YYYY-MM-DD HH:MM:SS"
182 // or with ASN.1 TLV. Try ASN.1 first, then fall back to plain text.
183 const auto* bytes = reinterpret_cast<const uint8_t*>(rawResponse.data());
184 std::string decoded;
185 std::string decodeErr;
186
187 if (asn_date::decodeASN1Date(bytes, rawResponse.size(), decoded, &decodeErr))
188 {
189 result.success = true;
190 result.datetime = decoded;
191 }
192 else
193 {
194 // Fallback: treat the cleaned text as a datetime string
195 std::string cleaned = cleanRelayText(rawResponse);
196 if (!cleaned.empty())
197 {
198 result.success = true;
199 result.datetime = cleaned;
200 }
201 else
202 {
203 result.error = "Cannot parse DATE response: " + decodeErr;
204 }
205 }
206
207 return result;
208 }

218 DateResult syncRelayTime(const std::string& datetime)
219 {
220 std::lock_guard<std::mutex> lock(mutex_);
221 DateResult result;
222
223 if (!client_.isConnected())
224 {
225 result.error = "Relay not connected";
226 return result;
227 }
228
229 // Build ASN.1 encoded payload
230 auto payload = asn_date::encodeASN1Date(datetime);
231
232 // Send SETTIME command with the datetime string
233 // Relay expects: SETTIME YYYY-MM-DD HH:MM:SS
234 std::string cmd = "SETTIME " + datetime;
235 std::string rawResponse;
236 bool ok = client_.SendCmdReceiveData(cmd, rawResponse);
237
238 if (!ok)
239 {
240 result.error = "SETTIME command failed";
241 return result;
242 }
243
244 // Check for ACK in response
245 std::string upper = rawResponse;
246 std::transform(upper.begin(), upper.end(), upper.begin(), ::toupper);
247
248 if (upper.find("ACK") != std::string::npos
249 || upper.find("OK") != std::string::npos
250 || upper.find("=>") != std::string::npos)
251 {
252 result.success = true;
253 result.datetime = datetime;
254 }
255 else
256 {
257 result.error = "No ACK received from relay: " + rawResponse.substr(0, 120);
258 }
259
260 return result;
261 }

325 static std::string cleanRelayText(const std::string& raw)
326 {
327 // Look for a line containing a date-like pattern YYYY-MM-DD or DD/MM/YY
328 std::istringstream iss(raw);
329 std::string line;
330 while (std::getline(iss, line))
331 {
332 // Trim
333 auto start = line.find_first_not_of(" \t\r\n");
334 if (start == std::string::npos) continue;
335 auto end = line.find_last_not_of(" \t\r\n");
336 std::string trimmed = line.substr(start, end - start + 1);
337
338 // Skip echo of the command itself
339 if (trimmed.find("DATE") == 0 || trimmed.find("date") == 0)
340 continue;
341 // Skip prompts
342 if (trimmed.find("=>") != std::string::npos && trimmed.size() < 5)
343 continue;
344
345 // Accept if it contains digits and separators
346 bool hasDigit = false;
347 bool hasSep = false;
348 for (char c : trimmed)
349 {
350 if (std::isdigit(static_cast<unsigned char>(c))) hasDigit = true;
351 if (c == '-' || c == '/' || c == ':') hasSep = true;
352 }
353 if (hasDigit && hasSep)
354 return trimmed;
355 }
356 return {};
357 }
