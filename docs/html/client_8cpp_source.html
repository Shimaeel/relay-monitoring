45// ================= CONNECT =================
46bool TelnetClient::connectCheck(const std::string& host,
47 int port,
48 std::chrono::milliseconds timeout)
49{
50 try
51 {
52 io_.restart();
53 tcp::resolver resolver(io_);
54 boost::system::error_code ec;
55 auto endpoints = resolver.resolve(host, std::to_string(port), ec);
56 if (ec)
57 {
58 if (socket_.is_open())
59 socket_.close();
60 connected_ = false;
61 last_io_ok_ = false;
62 return false;
63 }
64
65 bool connected = false;
66 asio::steady_timer timer(io_);
67 timer.expires_after(timeout);
68
69 if (socket_.is_open())
70 socket_.close();
71
72 timer.async_wait([this](const boost::system::error_code& t_ec) {
73 if (!t_ec)
74 socket_.close();
75 });
76
77 asio::async_connect(
78 socket_,
79 endpoints,
80 [&](const boost::system::error_code& c_ec, const tcp::endpoint&) {
81 if (!c_ec)
82 connected = true;
83 ec = c_ec;
84 timer.cancel();
85 });
86
87 io_.run();
88
89 connected_ = connected;
90 last_io_ok_ = connected;
91 return connected;
92 }
93 catch (const std::exception& e)
94 {
95 std::cerr << "Connection failed: " << e.what() << std::endl;
96 if (socket_.is_open())
97 socket_.close();
98 connected_ = false;
99 last_io_ok_ = false;
100 return false;
101 }
102}

104// ================= GENERIC SEND =================
105bool TelnetClient::SendCmdReceiveData(const std::string& cmd,
106 std::string& outBuffer)
107{
108 if (!connected_)
109 {
110 // std::cout << "[DEBUG] SendCmdReceiveData: Not connected\n";
111 last_io_ok_ = false;
112 return false;
113 }
114
115 try
116 {
117 // std::cout << "[DEBUG] SendCmdReceiveData: Sending '" << cmd << "'\n";
118 outBuffer.clear();
119 last_response_.clear();
120
121 std::string fullCmd = cmd + "\r\n";
122 asio::write(socket_, asio::buffer(fullCmd));
123
124 auto start = std::chrono::steady_clock::now();
125 auto lastDataTime = start;
126 int readCount = 0;
127
128 // For SER command, use longer idle timeout to collect all data
129 bool isSERCmd = (cmd == "SER" || cmd == "ser");
130 auto idleTimeout = isSERCmd ? std::chrono::milliseconds(500) : std::chrono::milliseconds(200);
131
132 // Set socket to non-blocking for idle detection
133 socket_.non_blocking(true);
134
135 while (true)
136 {
137 std::array<char, 4096> data{}; // Larger buffer
138 boost::system::error_code ec;
139 std::size_t bytes = socket_.read_some(asio::buffer(data), ec);
140
141 if (ec == boost::asio::error::would_block)
142 {
143 // No data available, check if we should stop
144 auto now = std::chrono::steady_clock::now();
145
146 // If we have data and haven't received anything for a while, we're done
147 if (!outBuffer.empty() && (now - lastDataTime) > idleTimeout)
148 {
149 socket_.non_blocking(false);
150 last_io_ok_ = true;
151 return true;
152 }
153
154 // Check overall timeout
155 if (now - start > io_timeout_)
156 {
157 socket_.non_blocking(false);
158 last_io_ok_ = !outBuffer.empty(); // Success if we got some data
159 return last_io_ok_;
160 }
161
162 // Small sleep to avoid busy loop
163 std::this_thread::sleep_for(std::chrono::milliseconds(10));
164 continue;
165 }
166
167 if (ec)
168 {
169 socket_.non_blocking(false);
170 // std::cout << "[DEBUG] Read error: " << ec.message() << "\n";
171 last_io_ok_ = false;
172 return false;
173 }
174
175 if (bytes > 0)
176 {
177 readCount++;
178 lastDataTime = std::chrono::steady_clock::now();
179 outBuffer.append(data.data(), bytes);
180 last_response_ = outBuffer;
181 }
182
183 // Check for explicit completion markers
184 if (isResponseComplete(outBuffer))
185 {
186 socket_.non_blocking(false);
187 last_io_ok_ = true;
188 return true;
189 }
190
191 if (std::chrono::steady_clock::now() - start > io_timeout_)
192 {
193 socket_.non_blocking(false);
194 last_io_ok_ = !outBuffer.empty();
195 return last_io_ok_;
196 }
197 }
198 }
199 catch (const std::exception& e)
200 {
201 socket_.non_blocking(false);
202 std::cerr << "SendCmdReceiveData error: "
203 << e.what() << std::endl;
204 last_io_ok_ = false;
205 return false;
206 }
207}

251bool TelnetClient::isResponseComplete(const std::string& buffer) const
252{
253 if (buffer.find("SER Response Complete") != std::string::npos)
254 return true;
255
256 // For SER data, look for the actual command prompt at end of response
257 // The relay shows "=>" as the prompt after SER data
258 return endsWithPrompt(buffer);
259}

261bool TelnetClient::endsWithPrompt(const std::string& buffer)
262{
263 // Check last 50 characters for actual command prompt
264 // Looking for patterns like "\n=>" or "Level 1\n=>"
265 size_t len = buffer.length();
266 if (len < 3) return false;
267
268 // Look at last 50 characters
269 size_t start = (len > 50) ? len - 50 : 0;
270 std::string tail = buffer.substr(start);
271
272 // Look for "=>" prompt which indicates end of relay response
273 // Must be at end of a line or end of buffer
274 size_t promptPos = tail.rfind("=>");
275 if (promptPos != std::string::npos)
276 {
277 // Check if it's at the end (possibly followed by whitespace)
278 size_t afterPrompt = promptPos + 2;
279 while (afterPrompt < tail.length())
280 {
281 char c = tail[afterPrompt];
282 if (c != ' ' && c != '\r' && c != '\n' && c != '\t')
283 break;
284 afterPrompt++;
285 }
286 if (afterPrompt >= tail.length())
287 return true;
288 }
289
290 // Also check for other common prompts at end
291 // But NOT just ":" which appears in timestamps
292 if (tail.length() >= 2)
293 {
294 std::string lastTwo = tail.substr(tail.length() - 2);
295 if (lastTwo == "> " || lastTwo == ">\r" || lastTwo == ">\n")
296 return true;
297 if (lastTwo == "# " || lastTwo == "#\r" || lastTwo == "#\n")
298 return true;
299 if (lastTwo == "$ " || lastTwo == "$\r" || lastTwo == "$\n")
300 return true;
301 }
302
303 // Check for "? " prompt (for questions)
304 if (tail.length() >= 2)
305 {
306 std::string lastTwo = tail.substr(tail.length() - 2);
307 if (lastTwo == "? ")
308 return true;
309 }
310
311 return false;
312}
