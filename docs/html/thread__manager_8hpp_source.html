71template<typename T>
72class ThreadSafeQueue
73{
74 std::queue<T> queue_;
75 mutable std::mutex mutex_;
76 std::condition_variable cv_;
77
78public:
83 void push(T item)
84 {
85 {
86 std::lock_guard<std::mutex> lock(mutex_);
87 queue_.push(std::move(item));
88 }
89 cv_.notify_one();
90 }
91
98 bool pop(T& item, std::atomic<bool>& stop_flag)
99 {
100 std::unique_lock<std::mutex> lock(mutex_);
101 cv_.wait(lock, [this, &stop_flag]() {
102 return !queue_.empty() || stop_flag.load();
103 });
104
105 if (stop_flag.load() && queue_.empty())
106 return false;
107
108 item = std::move(queue_.front());
109 queue_.pop();
110 return true;
111 }
112
117 bool empty() const
118 {
119 std::lock_guard<std::mutex> lock(mutex_);
120 return queue_.empty();
121 }
122
127 size_t size() const
128 {
129 std::lock_guard<std::mutex> lock(mutex_);
130 return queue_.size();
131 }
132
136 void notify_all()
137 {
138 cv_.notify_all();
139 }
140};

193 {
194 stop_flag_ = false;

196 std::cout << "[DB Writer] Thread started\n";

198 while (!stop_flag_.load())
199 {
200 SERRecord record;
201 if (write_queue_.pop(record, stop_flag_))

203 std::lock_guard<std::mutex> lock(db_mutex_);
204 if (db_.insertRecord(record))

211 std::cout << "[DB Writer] Thread stopped\n";
212 });
