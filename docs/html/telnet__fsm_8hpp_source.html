361struct PollSerAction
362{
370 template <class Event>
371 void operator()(const Event&, TelnetClient& client, SERDatabase& db, SharedRingBuffer& ring) const
372 {
373 client.clearLastResponse();
374 std::string buffer;
375 client.SendCmdReceiveData("SER", buffer);
376
377 // Parse and store SER records in database
378 auto records = parseSERResponse(buffer);
379
380 if (!records.empty())
381 {
382 int inserted = db.insertRecords(records);
383 std::cout << "[SER] Parsed " << records.size() << " records, stored " << inserted << " in database\n";
384 std::cout << "Records size before TLV encode: "
385 << records.size() << std::endl;
386
387 auto payload = asn_tlv::encodeSerRecordsToTlv(records);
388 if (!payload.empty())
389 {
390 ring.write(payload.data(), payload.size());
391 }
392 }
393 else
394 {
395 std::cout << "[SER] No records parsed - check response format!\n";
396 }
397 }

602inline bool hasPrompt(const std::string& buffer)
603{
604 // Check last 30 characters for prompt, ignoring telnet control codes
605 size_t len = buffer.length();
606 // std::cout << "[DEBUG hasPrompt] Buffer length: " << len << "\n";
607 if (len == 0) return false;
608
609 // Look at last 30 characters for a prompt character
610 size_t start = (len > 30) ? len - 30 : 0;
611 // std::cout << "[DEBUG hasPrompt] Checking from pos " << start << " to " << len << "\n";
612 for(size_t i = len; i > start; --i) {
613 char c = buffer[i-1];
614 if (c == '>' || c == '#' || c == '$' || c == ':' || c == '?') {
615 // std::cout << "[DEBUG hasPrompt] Found '" << c << "' at pos " << (i-1) << "\n";
616 return true;
617 }
618 }
619 // std::cout << "[DEBUG hasPrompt] No prompt found\n";
620 return false;
621}

902struct TelnetFSM
903{
914 auto operator()() const
915 {
916 using namespace sml;
917
918 return make_transition_table(
919 // Entry / Exit
920 "Connecting"_s + on_entry<_> / ConnectAction{},
921 "Login_L1"_s + on_entry<_> / Login1Action{},
922 "Polling"_s + on_entry<_> / PollSerAction{},
923 "WaitingRetry"_s + on_entry<_> / RetryWaitAction{},
924 "Operational"_s + on_entry<_> / ResetRetryAction{},
925
926 // Transitions
927 *"Idle"_s + event<start_event> = "Connecting"_s,
928
929 "Connecting"_s + event<step_event> [ ConnectOkGuard{} ] = "Login_L1"_s,
930 "Connecting"_s + event<step_event> [ ConnectFailGuard{} ] = "Error"_s,
931
932 // Login with retry logic
933 "Login_L1"_s + event<step_event> [ Login1CompleteGuard{} ] = "Operational"_s,
934 "Login_L1"_s + event<step_event> [ CanRetryGuard{} ] = "WaitingRetry"_s,
935 "Login_L1"_s + event<step_event> [ MaxRetriesReachedGuard{} ] = "Error"_s,
936
937 // After waiting, retry connection
938 "WaitingRetry"_s + event<step_event> = "Connecting"_s,
939
940 "Operational"_s + event<step_event> = "Polling"_s,
941
942 // After successful poll, go to Done state (poll only once)
943 "Polling"_s + event<step_event> [ SerCompleteGuard{} ] = "Done"_s,
944 "Polling"_s + event<step_event> [ SerFailGuard{} ] = "Error"_s,
945
946 // Done state - stay here (no more polling)
947 "Done"_s + event<step_event> = "Done"_s,
948
949 // Safety
950 state<_> + event<unhandled_event> / on_unhandled = "Error"_s,
951 state<_> + unexpected_event<_> / on_unexpected = "Error"_s
952 );
953 }
954};
