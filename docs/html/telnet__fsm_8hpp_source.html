300struct PollSerAction
301{
309 template <class Event>
310 void operator()(const Event&, TelnetClient& client, SERDatabase& db, SharedRingBuffer& ring) const
311 {
312 client.clearLastResponse();
313 std::string buffer;
314 client.SendCmdReceiveData("SER", buffer);
315
316 // Parse and store SER records in database
317 auto records = parseSERResponse(buffer);
318
319 if (!records.empty())
320 {
321 int inserted = db.insertRecords(records);
322 std::cout << "[SER] Parsed " << records.size() << " records, stored " << inserted << " in database\n";
323 std::cout << "Records size before TLV encode: "
324 << records.size() << std::endl;
325
326 auto payload = asn_tlv::encodeSerRecordsToTlv(records);
327 if (!payload.empty())
328 {
329 ring.write(payload.data(), payload.size());
330 }
331 }
332 else
333 {
334 std::cout << "[SER] No records parsed - check response format!\n";
335 }
336 }

399inline bool hasPrompt(const std::string& buffer)
400{
401 // Check last 30 characters for prompt, ignoring telnet control codes
402 size_t len = buffer.length();
403 // std::cout << "[DEBUG hasPrompt] Buffer length: " << len << "\n";
404 if (len == 0) return false;
405
406 // Look at last 30 characters for a prompt character
407 size_t start = (len > 30) ? len - 30 : 0;
408 // std::cout << "[DEBUG hasPrompt] Checking from pos " << start << " to " << len << "\n";
409 for(size_t i = len; i > start; --i) {
410 char c = buffer[i-1];
411 if (c == '>' || c == '#' || c == '$' || c == ':' || c == '?') {
412 // std::cout << "[DEBUG hasPrompt] Found '" << c << "' at pos " << (i-1) << "\n";
413 return true;
414 }
415 }
416 // std::cout << "[DEBUG hasPrompt] No prompt found\n";
417 return false;
418}

699struct TelnetFSM
700{
711 auto operator()() const
712 {
713 using namespace sml;
714
715 return make_transition_table(
716 // Entry / Exit
717 "Connecting"_s + on_entry<_> / ConnectAction{},
718 "Login_L1"_s + on_entry<_> / Login1Action{},
719 "Polling"_s + on_entry<_> / PollSerAction{},
720 "WaitingRetry"_s + on_entry<_> / RetryWaitAction{},
721 "Operational"_s + on_entry<_> / ResetRetryAction{},
722
723 // Transitions
724 *"Idle"_s + event<start_event> = "Connecting"_s,
725
726 "Connecting"_s + event<step_event> [ ConnectOkGuard{} ] = "Login_L1"_s,
727 "Connecting"_s + event<step_event> [ ConnectFailGuard{} ] = "Error"_s,
728
729 // Login with retry logic
730 "Login_L1"_s + event<step_event> [ Login1CompleteGuard{} ] = "Operational"_s,
731 "Login_L1"_s + event<step_event> [ CanRetryGuard{} ] = "WaitingRetry"_s,
732 "Login_L1"_s + event<step_event> [ MaxRetriesReachedGuard{} ] = "Error"_s,
733
734 // After waiting, retry connection
735 "WaitingRetry"_s + event<step_event> = "Connecting"_s,
736
737 "Operational"_s + event<step_event> = "Polling"_s,
738
739 // After successful poll, go to Done state (poll only once)
740 "Polling"_s + event<step_event> [ SerCompleteGuard{} ] = "Done"_s,
741 "Polling"_s + event<step_event> [ SerFailGuard{} ] = "Error"_s,
742
743 // Done state - stay here (no more polling)
744 "Done"_s + event<step_event> = "Done"_s,
745
746 // Safety
747 state<_> + event<unhandled_event> / on_unhandled = "Error"_s,
748 state<_> + unexpected_event<_> / on_unexpected = "Error"_s
749 );
750 }
