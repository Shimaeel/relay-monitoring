387inline bool hasPrompt(const std::string& buffer)
388{
389 // Check last 30 characters for prompt, ignoring telnet control codes
390 size_t len = buffer.length();
391 // std::cout << "[DEBUG hasPrompt] Buffer length: " << len << "\n";
392 if (len == 0) return false;
393
394 // Look at last 30 characters for a prompt character
395 size_t start = (len > 30) ? len - 30 : 0;
396 // std::cout << "[DEBUG hasPrompt] Checking from pos " << start << " to " << len << "\n";
397 for(size_t i = len; i > start; --i) {
398 char c = buffer[i-1];
399 if (c == '>' || c == '#' || c == '$' || c == ':' || c == '?') {
400 // std::cout << "[DEBUG hasPrompt] Found '" << c << "' at pos " << (i-1) << "\n";
401 return true;
402 }
403 }
404 // std::cout << "[DEBUG hasPrompt] No prompt found\n";
405 return false;
406}

686struct TelnetFSM
687{
698 auto operator()() const
699 {
700 using namespace sml;
701
702 return make_transition_table(
703 // Entry / Exit
704 "Connecting"_s + on_entry<_> / ConnectAction{},
705 "Login_L1"_s + on_entry<_> / Login1Action{},
706 "Polling"_s + on_entry<_> / PollSerAction{},
707 "WaitingRetry"_s + on_entry<_> / RetryWaitAction{},
708 "Operational"_s + on_entry<_> / ResetRetryAction{},
709
710 // Transitions
711 *"Idle"_s + event<start_event> = "Connecting"_s,
712
713 "Connecting"_s + event<step_event> [ ConnectOkGuard{} ] = "Login_L1"_s,
714 "Connecting"_s + event<step_event> [ ConnectFailGuard{} ] = "Error"_s,
715
716 // Login with retry logic
717 "Login_L1"_s + event<step_event> [ Login1CompleteGuard{} ] = "Operational"_s,
718 "Login_L1"_s + event<step_event> [ CanRetryGuard{} ] = "WaitingRetry"_s,
719 "Login_L1"_s + event<step_event> [ MaxRetriesReachedGuard{} ] = "Error"_s,
720
721 // After waiting, retry connection
722 "WaitingRetry"_s + event<step_event> = "Connecting"_s,
723
724 "Operational"_s + event<step_event> = "Polling"_s,
725
726 // After successful poll, go to Done state (poll only once)
727 "Polling"_s + event<step_event> [ SerCompleteGuard{} ] = "Done"_s,
728 "Polling"_s + event<step_event> [ SerFailGuard{} ] = "Error"_s,
729
730 // Done state - stay here (no more polling)
731 "Done"_s + event<step_event> = "Done"_s,
732
733 // Safety
734 state<_> + event<unhandled_event> / on_unhandled = "Error"_s,
735 state<_> + unexpected_event<_> / on_unexpected = "Error"_s
736 );
737 }
738};
