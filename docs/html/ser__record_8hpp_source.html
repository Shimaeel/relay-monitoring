212inline std::string sanitizeSerLine(const std::string& line)
213{
214 std::string out;
215 out.reserve(line.size());
216
217 for (std::size_t i = 0; i < line.size(); ++i)
218 {
219 unsigned char c = static_cast<unsigned char>(line[i]);
220
221 // Strip Telnet IAC sequences (0xFF cmd opt)
222 if (c == 0xFF)
223 {
224 if (i + 1 < line.size() && static_cast<unsigned char>(line[i + 1]) == 0xFF)
225 {
226 ++i;
227 continue;
228 }
229 if (i + 2 < line.size())
230 {
231 i += 2;
232 continue;
233 }
234 break;
235 }
236
237 // Strip ANSI escape sequences (ESC [ ...)
238 if (c == 0x1B)
239 {
240 if (i + 1 < line.size() && line[i + 1] == '[')
241 {
242 i += 2;
243 while (i < line.size())
244 {
245 unsigned char esc = static_cast<unsigned char>(line[i]);
246 if (esc >= 0x40 && esc <= 0x7E)
247 break;
248 ++i;
249 }
250 continue;
251 }
252 continue;
253 }
254
255 // Keep printable ASCII and tabs; drop other control chars
256 if (c >= 0x20 || c == '\t')
257 out.push_back(static_cast<char>(c));
258 }
259
260 return out;
261}

293inline std::vector<SERRecord> parseSERResponse(const std::string& response)
294{
295 std::vector<SERRecord> records;
296 std::istringstream stream(response);
297 std::string line;
298
299 while (std::getline(stream, line))
300 {
301 line = sanitizeSerLine(line);
302
303 // Remove \r if present
304 if (!line.empty() && line.back() == '\r')
305 line.pop_back();
306
307 // Skip empty lines and header lines
308 if (line.empty() || line.find('#') == 0 || line.find("Date") != std::string::npos)
309 continue;
310
311 // Trim leading whitespace
312 size_t start = line.find_first_not_of(" \t");
313 if (start == std::string::npos)
314 continue;
315
316 // Try to parse: "45 02/14/22 12:47:19.970 Power loss"
317 // Format: number date time element [state]
318
319 // Check if line starts with a number (SER record)
320 size_t digitPos = line.find_first_of("0123456789", start);
321 if (digitPos == std::string::npos)
322 continue;
323 if (digitPos != start)
324 line = line.substr(digitPos);
325
326 // Parse using position-based extraction
327 std::istringstream lineStream(line);
328
329 int recordNum;
330 std::string date, time, element, state;
331
332 // Read record number, date, time
333 if (!(lineStream >> recordNum >> date >> time))
334 continue;
335
336 // Validate date format: MM/DD/YY or MM/DD/YYYY (must have slashes at pos 2 and 5)
337 if (date.size() < 8 || date[2] != '/' || date[5] != '/')
338 continue;
339
340 // Validate time format: must contain at least one colon (HH:MM:SS.xxx)
341 if (time.find(':') == std::string::npos)
342 continue;
343
344 // Read rest of line as element + optional state
345 std::string rest;
346 std::getline(lineStream, rest);
347
348 // Trim leading whitespace from rest
349 size_t restStart = rest.find_first_not_of(" \t");
350 if (restStart != std::string::npos)
351 rest = rest.substr(restStart);
352 else
353 rest = "";
354
355 // Check if last word is a state (Asserted/Deasserted)
356 size_t lastSpace = rest.rfind(' ');
357 if (lastSpace != std::string::npos)
358 {
359 std::string lastWord = rest.substr(lastSpace + 1);
360 // Trim trailing whitespace from lastWord
361 size_t endPos = lastWord.find_last_not_of(" \t\r\n");
362 if (endPos != std::string::npos)
363 lastWord = lastWord.substr(0, endPos + 1);
364
365 if (lastWord == "Asserted" || lastWord == "Deasserted")
366 {
367 state = lastWord;
368 element = rest.substr(0, lastSpace);
369 // Trim trailing whitespace from element
370 size_t elemEnd = element.find_last_not_of(" \t");
371 if (elemEnd != std::string::npos)
372 element = element.substr(0, elemEnd + 1);
373 }
374 else
375 {
376 element = rest;
377 state = "";
378 }
379 }
380 else
381 {
382 element = rest;
383 state = "";
384 }
385
386 if (element.empty())
387 continue;
388
389 // Keep original date format from relay (MM/DD/YY)
390 std::string formattedDate = date;
391
392 SERRecord record;
393 record.record_id = std::to_string(recordNum); // Keep original # from relay
394 record.timestamp = formattedDate + " " + time;
395 record.status = state; // Keep original state (Asserted/Deasserted/empty)
396 record.description = element;
397 records.push_back(record);
398 }
399
400 return records;
401}
