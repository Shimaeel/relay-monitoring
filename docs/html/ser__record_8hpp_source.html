174inline std::string sanitizeSerLine(const std::string& line)
175{
176 std::string out;
177 out.reserve(line.size());
178
179 for (std::size_t i = 0; i < line.size(); ++i)
180 {
181 unsigned char c = static_cast<unsigned char>(line[i]);
182
183 // Strip Telnet IAC sequences (0xFF cmd opt)
184 if (c == 0xFF)
185 {
186 if (i + 1 < line.size() && static_cast<unsigned char>(line[i + 1]) == 0xFF)
187 {
188 ++i;
189 continue;
190 }
191 if (i + 2 < line.size())
192 {
193 i += 2;
194 continue;
195 }
196 break;
197 }
198
199 // Strip ANSI escape sequences (ESC [ ...)
200 if (c == 0x1B)
201 {
202 if (i + 1 < line.size() && line[i + 1] == '[')
203 {
204 i += 2;
205 while (i < line.size())
206 {
207 unsigned char esc = static_cast<unsigned char>(line[i]);
208 if (esc >= 0x40 && esc <= 0x7E)
209 break;
210 ++i;
211 }
212 continue;
213 }
214 continue;
215 }
216
217 // Keep printable ASCII and tabs; drop other control chars
218 if (c >= 0x20 || c == '\t')
219 out.push_back(static_cast<char>(c));
220 }
221
222 return out;
223}

225inline std::vector<SERRecord> parseSERResponse(const std::string& response)
226{
227 std::vector<SERRecord> records;
228 std::istringstream stream(response);
229 std::string line;
230
231 while (std::getline(stream, line))
232 {
233 line = sanitizeSerLine(line);
234
235 // Remove \r if present
236 if (!line.empty() && line.back() == '\r')
237 line.pop_back();
238
239 // Skip empty lines and header lines
240 if (line.empty() || line.find('#') == 0 || line.find("Date") != std::string::npos)
241 continue;
242
243 // Trim leading whitespace
244 size_t start = line.find_first_not_of(" \t");
245 if (start == std::string::npos)
246 continue;
247
248 // Try to parse: "45 02/14/22 12:47:19.970 Power loss"
249 // Format: number date time element [state]
250
251 // Check if line starts with a number (SER record)
252 size_t digitPos = line.find_first_of("0123456789", start);
253 if (digitPos == std::string::npos)
254 continue;
255 if (digitPos != start)
256 line = line.substr(digitPos);
257
258 // Parse using position-based extraction
259 std::istringstream lineStream(line);
260
261 int recordNum;
262 std::string date, time, element, state;
263
264 // Read record number, date, time
265 if (!(lineStream >> recordNum >> date >> time))
266 continue;
267
268 // Read rest of line as element + optional state
269 std::string rest;
270 std::getline(lineStream, rest);
271
272 // Trim leading whitespace from rest
273 size_t restStart = rest.find_first_not_of(" \t");
274 if (restStart != std::string::npos)
275 rest = rest.substr(restStart);
276 else
277 rest = "";
278
279 // Check if last word is a state (Asserted/Deasserted)
280 size_t lastSpace = rest.rfind(' ');
281 if (lastSpace != std::string::npos)
282 {
283 std::string lastWord = rest.substr(lastSpace + 1);
284 // Trim trailing whitespace from lastWord
285 size_t endPos = lastWord.find_last_not_of(" \t\r\n");
286 if (endPos != std::string::npos)
287 lastWord = lastWord.substr(0, endPos + 1);
288
289 if (lastWord == "Asserted" || lastWord == "Deasserted")
290 {
291 state = lastWord;
292 element = rest.substr(0, lastSpace);
293 // Trim trailing whitespace from element
294 size_t elemEnd = element.find_last_not_of(" \t");
295 if (elemEnd != std::string::npos)
296 element = element.substr(0, elemEnd + 1);
297 }
298 else
299 {
300 element = rest;
301 state = "";
302 }
303 }
304 else
305 {
306 element = rest;
307 state = "";
308 }
309
310 if (element.empty())
311 continue;
312
313 // Keep original date format from relay (MM/DD/YY)
314 std::string formattedDate = date;
315
316 SERRecord record;
317 record.record_id = std::to_string(recordNum); // Keep original # from relay
318 record.timestamp = formattedDate + " " + time;
319 record.status = state; // Keep original state (Asserted/Deasserted/empty)
320 record.description = element;
321 records.push_back(record);
322 }
323
324 return records;
325}
