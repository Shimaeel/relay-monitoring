174inline std::vector<SERRecord> parseSERResponse(const std::string& response)
175{
176 std::vector<SERRecord> records;
177 std::istringstream stream(response);
178 std::string line;
179
180 while (std::getline(stream, line))
181 {
182 // Remove \r if present
183 if (!line.empty() && line.back() == '\r')
184 line.pop_back();
185
186 // Skip empty lines and header lines
187 if (line.empty() || line.find('#') == 0 || line.find("Date") != std::string::npos)
188 continue;
189
190 // Trim leading whitespace
191 size_t start = line.find_first_not_of(" \t");
192 if (start == std::string::npos)
193 continue;
194
195 // Try to parse: "45 02/14/22 12:47:19.970 Power loss"
196 // Format: number date time element [state]
197
198 // Check if line starts with a number (SER record)
199 if (!std::isdigit(line[start]))
200 continue;
201
202 // Parse using position-based extraction
203 std::istringstream lineStream(line);
204
205 int recordNum;
206 std::string date, time, element, state;
207
208 // Read record number, date, time
209 if (!(lineStream >> recordNum >> date >> time))
210 continue;
211
212 // Read rest of line as element + optional state
213 std::string rest;
214 std::getline(lineStream, rest);
215
216 // Trim leading whitespace from rest
217 size_t restStart = rest.find_first_not_of(" \t");
218 if (restStart != std::string::npos)
219 rest = rest.substr(restStart);
220 else
221 rest = "";
222
223 // Check if last word is a state (Asserted/Deasserted)
224 size_t lastSpace = rest.rfind(' ');
225 if (lastSpace != std::string::npos)
226 {
227 std::string lastWord = rest.substr(lastSpace + 1);
228 // Trim trailing whitespace from lastWord
229 size_t endPos = lastWord.find_last_not_of(" \t\r\n");
230 if (endPos != std::string::npos)
231 lastWord = lastWord.substr(0, endPos + 1);
232
233 if (lastWord == "Asserted" || lastWord == "Deasserted")
234 {
235 state = lastWord;
236 element = rest.substr(0, lastSpace);
237 // Trim trailing whitespace from element
238 size_t elemEnd = element.find_last_not_of(" \t");
239 if (elemEnd != std::string::npos)
240 element = element.substr(0, elemEnd + 1);
241 }
242 else
243 {
244 element = rest;
245 state = "";
246 }
247 }
248 else
249 {
250 element = rest;
251 state = "";
252 }
253
254 if (element.empty())
255 continue;
256
257 // Keep original date format from relay (MM/DD/YY)
258 std::string formattedDate = date;
259
260 SERRecord record;
261 record.record_id = std::to_string(recordNum); // Keep original # from relay
262 record.timestamp = formattedDate + " " + time;
263 record.status = state; // Keep original state (Asserted/Deasserted/empty)
264 record.description = element;
265 records.push_back(record);
266 }
267
268 return records;
269}
