70inline std::string escapeJson(const std::string& value)
71{
72 std::string out;
73 out.reserve(value.size() + 8U);
74 for (char ch : value)
75 {
76 if (ch == '"' || ch == '\\')
77 {
78 out.push_back('\\');
79 }
80 out.push_back(ch);
81 }
82 return out;
83}

101inline std::string recordsToJsonTable(const std::vector<SERRecord>& records)
102{
103 std::string json = "[\n";
104
105 for (std::size_t i = 0; i < records.size(); ++i)
106 {
107 const auto& rec = records[i];
108 std::string date = rec.timestamp;
109 std::string time;
110
111 std::size_t spacePos = rec.timestamp.find(' ');
112 if (spacePos != std::string::npos)
113 {
114 date = rec.timestamp.substr(0, spacePos);
115 time = rec.timestamp.substr(spacePos + 1U);
116 }
117
118 json += " {\n";
119 json += " \"sno\": ";
120 try
121 {
122 int value = std::stoi(rec.record_id);
123 json += std::to_string(value);
124 }
125 catch (...)
126 {
127 json += "\"" + escapeJson(rec.record_id) + "\"";
128 }
129 json += ",\n";
130 json += " \"date\": \"" + escapeJson(date) + "\",\n";
131 json += " \"time\": \"" + escapeJson(time) + "\",\n";
132 json += " \"element\": \"" + escapeJson(rec.description) + "\",\n";
133 json += " \"state\": \"" + escapeJson(rec.status) + "\"\n";
134 json += " }";
135
136 if (i + 1U < records.size())
137 json += ",";
138
139 json += "\n";
140 }
141
142 json += "]";
143 return json;
144}
