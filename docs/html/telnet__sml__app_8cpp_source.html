201 void executeCommand(const std::string& cmd)
202 {
203 std::cout << "[Reception] executeCommand called for: " << cmd << "\n";
204
205 for (int attempt = 1; attempt <= MAX_RETRIES; ++attempt)
206 {
207 if (stop_flag_.load())
208 return;
209
210 // --- Connection + Login ---
211 if (!ensureConnected())
212 {
213 std::cout << "[Reception] Connection failed (attempt " << attempt
214 << "/" << MAX_RETRIES << ")\n";
215 connected_ = false;
216 logged_in_ = false;
217
218 if (attempt < MAX_RETRIES)
219 {
220 std::cout << "[Reception] Retrying in " << RETRY_DELAY_SEC << "s...\n";
221 // Interruptible sleep — wake up early if app is stopping
222 for (int s = 0; s < RETRY_DELAY_SEC && !stop_flag_.load(); ++s)
223 std::this_thread::sleep_for(std::chrono::seconds(1));
224 }
225 continue;
226 }
227
228 // --- Send Command ---
229 std::string response;
230 std::cout << "[Reception] Sending command to relay (attempt " << attempt
231 << "/" << MAX_RETRIES << ")...\n";
232 bool ok = client_.SendCmdReceiveData(cmd, response);
233
234 std::cout << "[Reception] SendCmdReceiveData returned: " << (ok ? "OK" : "FAIL")
235 << ", response size: " << response.size() << "\n";
236
237 if (ok && !response.empty())
238 {
239 // Success — push to ring buffer
240 bool pushed = rawBuffer_.push({cmd, response});
241 std::cout << "[Reception] Pushed to ring buffer: " << (pushed ? "OK" : "OVERWRITE")
242 << ", " << response.size() << " bytes for '" << cmd << "'\n";
243 std::cout << "[Reception] Ring buffer size now: " << rawBuffer_.size() << "\n";
244 return; // Done — exit retry loop
245 }
246
247 // --- Command failed ---
248 std::cout << "[Reception] Command '" << cmd << "' failed or empty response (attempt "
249 << attempt << "/" << MAX_RETRIES << ")\n";
250 connected_ = false;
251 logged_in_ = false;
252
253 if (attempt < MAX_RETRIES)
254 {
255 std::cout << "[Reception] Retrying in " << RETRY_DELAY_SEC << "s...\n";
256 for (int s = 0; s < RETRY_DELAY_SEC && !stop_flag_.load(); ++s)
257 std::this_thread::sleep_for(std::chrono::seconds(1));
258 }
259 }
260
261 std::cout << "[Reception] All " << MAX_RETRIES << " attempts failed for '" << cmd
262 << "', will retry on next poll cycle\n";
263 }

275 void runLoop()
276 {
277 std::cout << "[Reception] Worker thread started\n";
278 while (!stop_flag_.load())
279 {
280 std::string cmd;
281 {
282 std::unique_lock<std::mutex> lock(queue_mutex_);
283 std::cout << "[Reception] Waiting for command in queue...\n";
284 queue_cv_.wait(lock, [&]() {
285 return stop_flag_.load() || !command_queue_.empty();
286 });
287
288 if (stop_flag_.load())
289 break;
290
291 if (command_queue_.empty())
292 continue;
293
294 cmd = std::move(command_queue_.front());
295 command_queue_.pop_front();
296 std::cout << "[Reception] Got command from queue: " << cmd << "\n";
297 }
298
299 executeCommand(cmd);
300 }
301 std::cout << "[Reception] Worker thread exiting\n";
302 }

440class ProcessingWorker
441{
442 std::atomic<bool> stop_flag_{false};

461 void runLoop()
462 {
463 std::cout << "[Processing] Worker thread started\n";
464 while (!stop_flag_.load() && app_running_.load())
465 {
466 std::cout << "[Processing] Waiting for data from ring buffer...\n";
467 auto msg = rawBuffer_.waitPop(stop_flag_);
468 if (!msg)
469 {
470 std::cout << "[Processing] waitPop returned empty (stop signal?)\n";
471 continue;
472 }
473
474 std::cout << "[Processing] Received message: cmd='" << msg->command
475 << "', response size=" << msg->response.size() << " bytes\n";
476
477 // Only process SER command responses through the SER pipeline
478 if (msg->command != "SER")
479 {
480 // Non-SER command (e.g. FIL DIR) — broadcast raw text response to all WS clients
481 std::cout << "[Processing] Non-SER command '" << msg->command
482 << "', broadcasting text response (" << msg->response.size() << " bytes)\n";
483 wsServer_.broadcastText(msg->response);
484 continue;
485 }
486
487 // Parse SER response
488 std::cout << "[Processing] Parsing SER response...\n";
489 auto records = parseSERResponse(msg->response);
490 if (records.empty())
491 {
492 std::cout << "[Processing] No records parsed from response. First 200 chars:\n";
493 std::cout << msg->response.substr(0, 200) << "\n";
494 continue;
495 }
496
497 std::cout << "[Processing] Parsed " << records.size() << " SER records\n";
498
499 // Step 1: SQLite (Persistent Storage)
500 int inserted = db_.insertRecords(records);
501 std::cout << "[Processing] SQLite: Stored " << inserted << " new records\n";
502
503 // Step 2: WebSocket Server (After DB Write) - always broadcast so UI stays current
504 // Even when inserted==0 (all duplicates), connected browsers need the data
505 // (e.g. browser connected after app restart with existing DB)
506 wsServer_.broadcast(records);
507 std::cout << "[Processing] WebSocket: Broadcast " << records.size() << " records to clients\n";
508
509 // Write to shared memory for JSON file writer (only on new data)
510 if (inserted > 0)
511 {
512 auto payload = asn_tlv::encodeSerRecordsToTlv(records);
513 if (!payload.empty())
514 {
515 shmRing_.write(payload.data(), payload.size());
516 }
517 }
518 }
519 std::cout << "[Processing] Worker thread exiting\n";
520 }

600class SharedSerReader
601{
602 SharedRingBuffer& ring_;
603 std::string output_path_;
604 std::atomic<bool> stop_flag_{false};
605 std::thread worker_thread_;
606 SharedRingBuffer::ReaderId readerId_{SharedRingBuffer::kInvalidReader};
607
617 void runLoop()
618 {
619 std::vector<uint8_t> payload;
620 while (!stop_flag_.load())
621 {
622 if (!ring_.waitRead(readerId_, payload, stop_flag_))
623 continue;
624
625 std::vector<SERRecord> records;
626 std::string error;
627 if (!asn_tlv::decodeSerRecordsFromTlv(payload.data(), payload.size(), records, &error))
628 {
629 std::cout << "[SHM] Decode error: " << error << "\n";
630 continue;
631 }
632
633 std::string json = recordsToJsonTable(records);
634 std::string writeError;
635 if (!writeJsonToFile(output_path_, json, &writeError))
636 {
637 std::cout << "[SHM] JSON write error: " << writeError << "\n";
638 }
639 }
640 }

642public:
649 SharedSerReader(SharedRingBuffer& ring, std::string output_path)
650 : ring_(ring), output_path_(std::move(output_path))

652 }

657 void start()
658 {
659 stop_flag_ = false;
660 readerId_ = ring_.registerReader();
661 if (readerId_ == SharedRingBuffer::kInvalidReader)
662 {
663 std::cerr << "[SHM] Failed to register reader — max readers reached\n";
664 return;
665 }
666 worker_thread_ = std::thread([this]() { runLoop(); });
667 }
668
672 void stop()
673 {
674 stop_flag_ = true;
675 ring_.notifyAll();
676 if (worker_thread_.joinable())
677 worker_thread_.join();
678 if (readerId_ != SharedRingBuffer::kInvalidReader)
679 {
680 ring_.unregisterReader(readerId_);
681 readerId_ = SharedRingBuffer::kInvalidReader;
682 }
