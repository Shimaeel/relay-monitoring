113class SharedSerReader
114{
115 SharedRingBuffer& ring_;
116 std::string output_path_;
117 std::atomic<bool> stop_flag_{false};
118 std::thread worker_thread_;
119 SharedRingBuffer::ReaderId readerId_{SharedRingBuffer::kInvalidReader};
120
130 void runLoop()
131 {
132 std::vector<uint8_t> payload;
133 while (!stop_flag_.load())
134 {
135 if (!ring_.waitRead(readerId_, payload, stop_flag_))
136 continue;
137
138 std::vector<SERRecord> records;
139 std::string error;
140 if (!asn_tlv::decodeSerRecordsFromTlv(payload.data(), payload.size(), records, &error))
141 {
142 std::cout << "[SHM] Decode error: " << error << "\n";
143 continue;
144 }
145
146 std::string json = recordsToJsonTable(records);
147 std::string writeError;
148 if (!writeJsonToFile(output_path_, json, &writeError))
149 {
150 std::cout << "[SHM] JSON write error: " << writeError << "\n";
151 }
152 }
153 }

162 SharedSerReader(SharedRingBuffer& ring, std::string output_path)
163 : ring_(ring), output_path_(std::move(output_path))
164 {
165 }
166
170 void start()
171 {
172 stop_flag_ = false;
173 readerId_ = ring_.registerReader();
174 if (readerId_ == SharedRingBuffer::kInvalidReader)
175 {
176 std::cerr << "[SHM] Failed to register reader — max readers reached\n";
177 return;
178 }
179 worker_thread_ = std::thread([this]() { runLoop(); });
180 }
181
185 void stop()
186 {
187 stop_flag_ = true;
188 ring_.notifyAll();
189 if (worker_thread_.joinable())
190 worker_thread_.join();
191 if (readerId_ != SharedRingBuffer::kInvalidReader)
192 {
193 ring_.unregisterReader(readerId_);
194 readerId_ = SharedRingBuffer::kInvalidReader;
195 }
196 }

274 static std::string extractJsonField(const std::string& json, const std::string& fieldName)
275 {
276 const std::string key = "\"" + fieldName + "\"";
277 auto pos = json.find(key);
278 if (pos == std::string::npos) return "";
279 pos = json.find(':', pos + key.size());
280 if (pos == std::string::npos) return "";
281 pos = json.find('"', pos + 1);
282 if (pos == std::string::npos) return "";
283 auto end = json.find('"', pos + 1);
284 if (end == std::string::npos) return "";
285 return json.substr(pos + 1, end - pos - 1);
286 }

306 if (running)
307 return true;
308
309 app_running = true;
310
311 std::cout << "========================================\n";
312 std::cout << " Telnet-SML Multi-Relay Application\n";
313 std::cout << " (On-Demand Architecture v3.0)\n";
314 std::cout << "========================================\n\n";

323

333 }
334
335 // 3. Create RelayManager (no pipelines started yet — on-demand)
336 relayMgr = std::make_unique<RelayManager>(
337 serDb, wsServer, shmRing, app_running);
338 std::cout << "[RelayMgr] Initialized with " << relayMgr->getConfigs().size()
339 << " relay configurations\n";
340
341 // 4. Set up WS command handler — routes commands to the correct relay pipeline
342 // Browser sends: "relay_id:command" (e.g. "1:SER", "2:TAR 0", "2:EVE")
343 // Fallback: if no colon, try to route to first active relay
344 wsServer.setCommandHandler([this](const std::string& cmd) -> std::string {
345 auto colon = cmd.find(':');
346 if (colon != std::string::npos)
347 {
348 std::string relayId = cmd.substr(0, colon);
349 std::string realCmd = cmd.substr(colon + 1);
350 std::cout << "[WS→Relay] Routing '" << realCmd << "' to relay " << relayId << " via Command FSM\n";
351 return relayMgr->handleUserCommand(relayId, realCmd);
352 }
353 else
354 {
355 // No relay prefix — try first active relay
356 auto ids = relayMgr->getActiveRelayIds();
357 if (!ids.empty())
358 {
359 std::cout << "[WS→Relay] Routing '" << cmd << "' to relay " << ids.front() << " via Command FSM\n";
360 return relayMgr->handleUserCommand(ids.front(), cmd);
361 }
362 return "";
363 }
364 });
365
366 // 4b. Set up streaming command handler — FETCH_ALL_TAR loops TAR 0..N on the server
367 wsServer.setStreamCommandHandler(
368 [this](const std::string& cmd,
369 WebSocketSession::StreamingCallback sendFn,
370 std::atomic<bool>& abort) -> bool
371 {
372 // Parse relay prefix (e.g. "3:FETCH_ALL_TAR")
373 auto colon = cmd.find(':');
374 std::string relayId, realCmd;
375 if (colon != std::string::npos)
376 {
377 relayId = cmd.substr(0, colon);
378 realCmd = cmd.substr(colon + 1);
379 }
380 else
381 {
382 auto ids = relayMgr->getActiveRelayIds();
383 if (ids.empty()) return false;
384 relayId = ids.front();
385 realCmd = cmd;
386 }
387
388 if (realCmd != "FETCH_ALL_TAR")
389 return false; // not a streaming command — fall through
390
391 std::cout << "[WS→Relay] FETCH_ALL_TAR streaming for relay " << relayId << "\n";
392 int count = 0;
393
394 for (int i = 0; !abort.load(); ++i)
395 {
396 std::string response = relayMgr->handleUserCommand(
397 relayId, "TAR " + std::to_string(i));
398 if (response.empty())
399 break; // no more rows — relay returned nothing
400
401 // Stop when relay reports "Invalid Target"
402 if (response.find("Invalid Target") != std::string::npos)
403 break;
404
405 // Stream: "TAR_STREAM:<index>\n<response>"
406 sendFn("TAR_STREAM:" + std::to_string(i) + "\n" + response);
407 ++count;
408 }
409
410 // End marker — tells browser how many rows were found
411 sendFn("TAR_STREAM_END:" + std::to_string(count));
412 std::cout << "[WS→Relay] FETCH_ALL_TAR done — " << count << " rows streamed\n";
413 return true;
414 });
415
416 // 5. Set up action handler — relay lifecycle + time sync + password
417 wsServer.setActionHandler([this](const std::string& jsonMsg) -> std::string {
418 std::string action = extractJsonField(jsonMsg, "action");
419
420 // ── Relay lifecycle management ──
421 if (action == "start_relay")
422 {
423 std::string relayId = extractJsonField(jsonMsg, "relay_id");
424 std::cout << "[WS->Action] Starting relay " << relayId << "\n";
425 bool ok = relayMgr->startRelay(relayId);
426 return ok
427 ? "{\"action\":\"start_relay\",\"relay_id\":\"" + relayId + "\",\"status\":\"success\"}"
428 : "{\"action\":\"start_relay\",\"relay_id\":\"" + relayId + "\",\"status\":\"failed\",\"error\":\"Unknown relay or start failed\"}";
429 }
430
431 if (action == "stop_relay")
432 {
433 std::string relayId = extractJsonField(jsonMsg, "relay_id");
434 std::cout << "[WS->Action] Stopping relay " << relayId << "\n";
435 bool ok = relayMgr->stopRelay(relayId);
436 return ok
437 ? "{\"action\":\"stop_relay\",\"relay_id\":\"" + relayId + "\",\"status\":\"success\"}"
438 : "{\"action\":\"stop_relay\",\"relay_id\":\"" + relayId + "\",\"status\":\"failed\",\"error\":\"Relay not active\"}";
439 }
440
441 if (action == "relay_status")
442 {
443 std::string result = "{\"action\":\"relay_status\",\"active\":[";
444 auto ids = relayMgr->getActiveRelayIds();
445 for (size_t i = 0; i < ids.size(); ++i)
446 {
447 if (i > 0) result += ",";
448 result += "\"" + ids[i] + "\"";
449 }
450 result += "]}";
451 return result;
452 }
453
454 // ── Time sync (requires relay_id to know which relay) ──
455 if (action == "read_time" || action == "sync_time")
456 {
457 std::string relayId = extractJsonField(jsonMsg, "relay_id");
458 if (relayId.empty())
459 return "{\"action\":\"" + action + "\",\"status\":\"failed\",\"error\":\"Missing relay_id\"}";
460
461 auto* pipeline = relayMgr->getPipeline(relayId);
462 if (!pipeline)
463 return "{\"action\":\"" + action + "\",\"status\":\"failed\",\"error\":\"Relay not active\"}";
464
465 RelayService svc(pipeline->getClient());
466 TimeSyncManager tsm(svc);
467 return tsm.handleAction(action);
468 }
469
470 // ── Password change (requires relay_id to know which relay) ──
471 // TODO: Route to per-relay PasswordManager when available
472 if (action == "change_password")
473 {
474 std::cout << "[WS->Action] PasswordManager not yet per-relay\n";
475 return "{\"action\":\"change_password\",\"status\":\"failed\",\"error\":\"Password change requires per-relay support (coming soon)\"}";
476 }
477
478 std::cout << "[WS->Action] Unknown action: " << action << "\n";
479 return "{\"status\":\"failed\",\"error\":\"Unknown action\"}";
480 });
481
482 // 6. Start WebSocket server
483 if (!wsServer.start())
484 {
485 std::cerr << "Failed to start WebSocket server\n";
486 serDb.close();
487 return false;
488 }
489
490 // 7. Generic database WebSocket API on port 8766
491 dbApiServer = std::make_unique<WSDBServer>(serDb.getDbHandle(), 8766);
492 if (!dbApiServer->start())
493 std::cerr << "[WSDB] Warning: DB API server failed to start\n";
494
495 // 8. Polling callback — queue SER to ALL active relays
496 threadMgr.setPollingCallback([this]() {
497 if (!app_running.load())
498 return;
499
500 auto ids = relayMgr->getActiveRelayIds();
501 if (ids.empty())
502 return;
503
504 std::cout << "[Poller] Requesting SER from " << ids.size() << " active relay(s)...\n";
505 for (const auto& id : ids)
506 relayMgr->queueCommand(id, "SER");
507 });
508
509 // 9. Start support threads
510 shmReader.start();
511 threadMgr.startAll();
512
513 // ─── Summary ─────────────────────────────────────────────
514 std::cout << "\n========================================\n";
515 std::cout << " Multi-Relay On-Demand Architecture\n";
516 std::cout << " Configured relays: " << relayMgr->getConfigs().size() << "\n";
517 for (const auto& cfg : relayMgr->getConfigs())
518 {
519 std::cout << " [" << cfg.id << "] " << cfg.name
520 << " @ " << cfg.host << ":" << cfg.port
521 << " (" << cfg.substation << " / " << cfg.bay << ")\n";
522 }
523 std::cout << "\n Shared Services:\n";
524 std::cout << " - WebSocket Server: ws://localhost:8765\n";
525 std::cout << " - DB API Server: ws://localhost:8766\n";
526 std::cout << " - SQLite Database: ser_records.db\n";
527 std::cout << " - Poller: 2 min interval\n";
528 std::cout << " - JSON Writer: ui/data.json\n";
529 std::cout << "\n No relays active - waiting for browser commands.\n";
530 std::cout << " Send {\"action\":\"start_relay\",\"relay_id\":\"1\"} to connect.\n";
531 std::cout << "========================================\n";
532
533 running = true;
534 return true;
535 }
