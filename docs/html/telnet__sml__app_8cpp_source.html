185 void executeCommand(const std::string& cmd)
186 {
187 std::cout << "[Reception] executeCommand called for: " << cmd << "\n";
188
189 if (!ensureConnected())
190 {
191 std::cout << "[Reception] Not connected, cannot execute command\n";

196 std::cout << "[Reception] Sending command to relay...\n";
197 bool ok = client_.SendCmdReceiveData(cmd, response);
198
199 std::cout << "[Reception] SendCmdReceiveData returned: " << (ok ? "OK" : "FAIL")
200 << ", response size: " << response.size() << "\n";
201
202 if (ok && !response.empty())
203 {
204 // Non-blocking push to ring buffer - processing happens in separate thread
205 bool pushed = rawBuffer_.push({cmd, response});
206 std::cout << "[Reception] Pushed to ring buffer: " << (pushed ? "OK" : "OVERWRITE")
207 << ", " << response.size() << " bytes for '" << cmd << "'\n";
208 std::cout << "[Reception] Ring buffer size now: " << rawBuffer_.size() << "\n";

228 void runLoop()
229 {
230 std::cout << "[Reception] Worker thread started\n";
231 while (!stop_flag_.load())
232 {
233 std::string cmd;
234 {
235 std::unique_lock<std::mutex> lock(queue_mutex_);
236 std::cout << "[Reception] Waiting for command in queue...\n";
237 queue_cv_.wait(lock, [&]() {
238 return stop_flag_.load() || !command_queue_.empty();
239 });
240
241 if (stop_flag_.load())
242 break;
243
244 if (command_queue_.empty())
245 continue;
246
247 cmd = std::move(command_queue_.front());
248 command_queue_.pop_front();
249 std::cout << "[Reception] Got command from queue: " << cmd << "\n";
250 }
251
252 executeCommand(cmd);
253 }
254 std::cout << "[Reception] Worker thread exiting\n";
255 }

414 void runLoop()
415 {
416 std::cout << "[Processing] Worker thread started\n";
417 while (!stop_flag_.load() && app_running_.load())
418 {
419 std::cout << "[Processing] Waiting for data from ring buffer...\n";
420 auto msg = rawBuffer_.waitPop(stop_flag_);
421 if (!msg)
422 {
423 std::cout << "[Processing] waitPop returned empty (stop signal?)\n";
424 continue;
425 }
426
427 std::cout << "[Processing] Received message: cmd='" << msg->command
428 << "', response size=" << msg->response.size() << " bytes\n";
429
430 // Only process SER command responses
431 if (msg->command != "SER")
432 {
433 std::cout << "[Processing] Skipping non-SER command: " << msg->command << "\n";
434 continue;
435 }
436
437 // Parse SER response
438 std::cout << "[Processing] Parsing SER response...\n";
439 auto records = parseSERResponse(msg->response);
440 if (records.empty())
441 {
442 std::cout << "[Processing] No records parsed from response. First 200 chars:\n";
443 std::cout << msg->response.substr(0, 200) << "\n";
444 continue;
445 }
446
447 std::cout << "[Processing] Parsed " << records.size() << " SER records\n";
448
449 // Step 1: SQLite (Persistent Storage)
450 int inserted = db_.insertRecords(records);
451 std::cout << "[Processing] SQLite: Stored " << inserted << " new records\n";
452
453 // Step 2: WebSocket Server (After DB Write) - only broadcast on successful storage
454 if (inserted > 0)
455 {
456 wsServer_.broadcast(records);
457 std::cout << "[Processing] WebSocket: Broadcast " << records.size() << " records to clients\n";
458
459 // Write to shared memory for JSON file writer
460 auto payload = asn_tlv::encodeSerRecordsToTlv(records);
461 if (!payload.empty())
462 {
463 shmRing_.write(payload.data(), payload.size());
464 }
465 }
466 else
467 {
468 std::cout << "[Processing] No new records inserted, skipping WebSocket broadcast\n";
469 }
470 }
471 std::cout << "[Processing] Worker thread exiting\n";
472 }

569 void runLoop()
570 {
571 std::vector<uint8_t> payload;

574 if (!ring_.waitRead(readerId_, payload, stop_flag_))
575 continue;

582 continue;

586 std::string writeError;

590 }
591 }
