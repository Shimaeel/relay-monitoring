159 CREATE TABLE IF NOT EXISTS ser_records (
160 id INTEGER PRIMARY KEY AUTOINCREMENT,
161 record_id TEXT NOT NULL,
162 timestamp TEXT NOT NULL,
163 status TEXT NOT NULL,
164 description TEXT,
165 created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
166 UNIQUE(record_id, timestamp)
167 );
168 CREATE INDEX IF NOT EXISTS idx_record_id ON ser_records(record_id);
169 CREATE INDEX IF NOT EXISTS idx_status ON ser_records(status);
170 CREATE INDEX IF NOT EXISTS idx_timestamp ON ser_records(timestamp);
171 )";
172
173 char* errMsg = nullptr;
174 int rc = sqlite3_exec(db_, sql, nullptr, nullptr, &errMsg);
175
176 if (rc != SQLITE_OK)
177 {
178 last_error_ = "SQL error: " + std::string(errMsg);
179 sqlite3_free(errMsg);
180 return false;
181 }
182
183 return true;

186// ================= INSERT OPERATIONS =================
187
195 *
196 * @param record The SERRecord to insert. All four fields
197 * (record_id, timestamp, status, description) are stored.
198 *
199 * @return true Record inserted or already existed (duplicate skip).
200 * @return false Database not open, prepare failed, or step failed.
201 * Check getLastError() for details.
202 *
203 * @pre isOpen() == true
204 *
205 * @see insertRecords() Bulk insert with transaction wrapping.
206 * @see recordExists() Duplicate detection helper.
207 */
208bool SERDatabase::insertRecord(const SERRecord& record)
209{
210 if (!db_)
211 {
212 last_error_ = "Database not open";
213 return false;
214 }
215
216 // Skip if record already exists
217 if (recordExists(record.record_id, record.timestamp))
218 {
219 return true; // Not an error, just skip duplicate
220 }
221
222 const char* sql = "INSERT INTO ser_records (record_id, timestamp, status, description) "
223 "VALUES (?, ?, ?, ?);";
224
225 sqlite3_stmt* stmt;
226 int rc = sqlite3_prepare_v2(db_, sql, -1, &stmt, nullptr);
227
228 if (rc != SQLITE_OK)
229 {
230 last_error_ = "Failed to prepare statement: " + std::string(sqlite3_errmsg(db_));
231 return false;
232 }
233
234 sqlite3_bind_text(stmt, 1, record.record_id.c_str(), -1, SQLITE_TRANSIENT);
235 sqlite3_bind_text(stmt, 2, record.timestamp.c_str(), -1, SQLITE_TRANSIENT);
236 sqlite3_bind_text(stmt, 3, record.status.c_str(), -1, SQLITE_TRANSIENT);
237 sqlite3_bind_text(stmt, 4, record.description.c_str(), -1, SQLITE_TRANSIENT);
238
239 rc = sqlite3_step(stmt);
240 sqlite3_finalize(stmt);
241
242 if (rc != SQLITE_DONE)
243 {
244 last_error_ = "Failed to insert record: " + std::string(sqlite3_errmsg(db_));
245 return false;
246 }
247
248 return true;
249}
250
256 * record is inserted via insertRecord(), so duplicates are
257 * silently skipped.
258 *
259 * @param records Vector of SERRecord objects to insert.
260 *
261 * @return int Number of records successfully inserted (excludes duplicates).
262 * Returns 0 if the database is not open.
263 *
264 * @pre isOpen() == true
265 *
266 * @note The entire batch is committed even if some individual inserts
267 * are skipped as duplicates.
268 */
269int SERDatabase::insertRecords(const std::vector<SERRecord>& records)
270{
271 if (!db_)
272 {
273 last_error_ = "Database not open";
274 return 0;
275 }
276
277 int inserted = 0;
278
279 // Begin transaction for better performance
280 sqlite3_exec(db_, "BEGIN TRANSACTION;", nullptr, nullptr, nullptr);
281
282 for (const auto& record : records)
283 {
284 if (insertRecord(record))
285 {
286 inserted++;
287 }
288 }
289
290 sqlite3_exec(db_, "COMMIT;", nullptr, nullptr, nullptr);
291
292 return inserted;
293}

295// ================= QUERY OPERATIONS =================
296
298 * @brief Retrieve all SER records ordered by timestamp (most recent first).
299 *
300 * @details Executes `SELECT ... ORDER BY timestamp DESC` and maps each
301 * row into a SERRecord struct. Returns an empty vector if the
302 * database is not open or the query fails.
303 *
304 * @return std::vector<SERRecord> All stored records, newest first.
305 * Empty if database is closed or an error occurs.
306 *
307 * @pre isOpen() == true
308 *
309 * @see getRecordsByStatus() Filtered version of this query.
310 */
311std::vector<SERRecord> SERDatabase::getAllRecords()
312{
313 std::vector<SERRecord> records;
314
315 if (!db_)
316 {
317 last_error_ = "Database not open";
318 return records;
319 }
320
321 const char* sql = "SELECT record_id, timestamp, status, description FROM ser_records "
322 "ORDER BY timestamp DESC;";
323
324 sqlite3_stmt* stmt;
325 int rc = sqlite3_prepare_v2(db_, sql, -1, &stmt, nullptr);
326
327 if (rc != SQLITE_OK)
328 {
329 last_error_ = "Failed to prepare statement: " + std::string(sqlite3_errmsg(db_));
330 return records;
331 }
332
333 while (sqlite3_step(stmt) == SQLITE_ROW)
334 {
335 SERRecord record;
336 record.record_id = reinterpret_cast<const char*>(sqlite3_column_text(stmt, 0));
337 record.timestamp = reinterpret_cast<const char*>(sqlite3_column_text(stmt, 1));
338 record.status = reinterpret_cast<const char*>(sqlite3_column_text(stmt, 2));
339
340 const char* desc = reinterpret_cast<const char*>(sqlite3_column_text(stmt, 3));
341 record.description = desc ? desc : "";
342
343 records.push_back(record);
344 }
345
346 sqlite3_finalize(stmt);
347 return records;
348}

490bool SERDatabase::recordExists(const std::string& recordId, const std::string& timestamp)
491{
492 const char* sql = "SELECT 1 FROM ser_records WHERE record_id = ? AND timestamp = ? LIMIT 1;";
493
494 sqlite3_stmt* stmt;
495 int rc = sqlite3_prepare_v2(db_, sql, -1, &stmt, nullptr);
496
497 if (rc != SQLITE_OK)
498 {
499 return false;
500 }
501
502 sqlite3_bind_text(stmt, 1, recordId.c_str(), -1, SQLITE_TRANSIENT);
503 sqlite3_bind_text(stmt, 2, timestamp.c_str(), -1, SQLITE_TRANSIENT);
504
505 bool exists = (sqlite3_step(stmt) == SQLITE_ROW);
506 sqlite3_finalize(stmt);
507
508 return exists;

511// ================= EXPORT TO JSON =================
512
521 * to produce valid JSON output.
522 *
523 * @param filePath Destination file path (e.g. "data.json"). Created or
524 * overwritten if it already exists.
525 *
526 * @return true File written successfully.
527 * @return false Database not open or file cannot be opened for writing.
528 * Check getLastError() for details.
529 *
530 * @pre isOpen() == true
531 *
532 * @see SERJsonWriter Alternative streaming JSON writer.
533 */
534bool SERDatabase::exportToJSON(const std::string& filePath)
535{
536 if (!db_)
537 {
538 last_error_ = "Database not open";
539 return false;
540 }
541
542 std::vector<SERRecord> records = getAllRecords();
543
544 std::ofstream file(filePath);
545 if (!file.is_open())
546 {
547 last_error_ = "Cannot open file for writing: " + filePath;
548 return false;
549 }
550
551 file << "[\n";
552
553 int sno = 1;
554 for (size_t i = 0; i < records.size(); ++i)
555 {
556 const auto& rec = records[i];
557
558 // Parse timestamp to extract date and time
559 std::string date = rec.timestamp;
560 std::string time = "";
561
562 size_t spacePos = rec.timestamp.find(' ');
563 if (spacePos != std::string::npos)
564 {
565 date = rec.timestamp.substr(0, spacePos);
566 time = rec.timestamp.substr(spacePos + 1);
567 }
568
569 // Escape special characters in description
570 std::string element = rec.description;
571 for (size_t j = 0; j < element.length(); ++j)
572 {
573 if (element[j] == '"' || element[j] == '\\')
574 {
575 element.insert(j, "\\");
576 ++j;
577 }
578 }
579
580 file << " {\n";
581 file << " \"sno\": " << sno++ << ",\n";
582 file << " \"date\": \"" << date << "\",\n";
583 file << " \"time\": \"" << time << "\",\n";
584 file << " \"element\": \"" << element << "\",\n";
585 file << " \"state\": \"" << rec.status << "\"\n";
586 file << " }";
587
588 if (i < records.size() - 1)
589 file << ",";
590
591 file << "\n";
592 }
593
594 file << "]\n";
595 file.close();
596
597 return true;
598}
