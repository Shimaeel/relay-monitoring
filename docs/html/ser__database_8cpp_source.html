156bool SERDatabase::createTable()
157{
158 const char* sql = R"(
159 CREATE TABLE IF NOT EXISTS ser_records (
160 id INTEGER PRIMARY KEY AUTOINCREMENT,
161 relay_id TEXT NOT NULL DEFAULT '',
162 relay_name TEXT NOT NULL DEFAULT '',
163 record_id TEXT NOT NULL,
164 timestamp TEXT NOT NULL,
165 status TEXT NOT NULL,
166 description TEXT,
167 created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
168 UNIQUE(relay_id, record_id, timestamp)
169 );
170 CREATE INDEX IF NOT EXISTS idx_relay_id ON ser_records(relay_id);
171 CREATE INDEX IF NOT EXISTS idx_record_id ON ser_records(record_id);
172 CREATE INDEX IF NOT EXISTS idx_status ON ser_records(status);
173 CREATE INDEX IF NOT EXISTS idx_timestamp ON ser_records(timestamp);
174 )";
175
176 char* errMsg = nullptr;
177 int rc = sqlite3_exec(db_, sql, nullptr, nullptr, &errMsg);
178
179 if (rc != SQLITE_OK)
180 {
181 last_error_ = "SQL error: " + std::string(errMsg);
182 sqlite3_free(errMsg);
183 return false;
184 }
185
186 // Migration: add relay columns to existing databases that don't have them
187 sqlite3_exec(db_, "ALTER TABLE ser_records ADD COLUMN relay_id TEXT NOT NULL DEFAULT '';",
188 nullptr, nullptr, nullptr);
189 sqlite3_exec(db_, "ALTER TABLE ser_records ADD COLUMN relay_name TEXT NOT NULL DEFAULT '';",
190 nullptr, nullptr, nullptr);
191 sqlite3_exec(db_, "CREATE INDEX IF NOT EXISTS idx_relay_id ON ser_records(relay_id);",
192 nullptr, nullptr, nullptr);
193
194 return true;

197// ================= INSERT OPERATIONS =================
198
203 * `(record_id, timestamp)`. If the record already exists it
204 * returns true (skip, not an error). Otherwise it prepares an
205 * INSERT statement with parameter binding for safety.
206 *
207 * @param record The SERRecord to insert. All four fields
208 * (record_id, timestamp, status, description) are stored.
209 *
210 * @return true Record inserted or already existed (duplicate skip).
211 * @return false Database not open, prepare failed, or step failed.
212 * Check getLastError() for details.
213 *
214 * @pre isOpen() == true
215 *
216 * @see insertRecords() Bulk insert with transaction wrapping.
217 * @see recordExists() Duplicate detection helper.
218 */
219bool SERDatabase::insertRecord(const SERRecord& record)
220{
221 if (!db_)
222 {
223 last_error_ = "Database not open";
224 return false;
225 }
226
227 // Skip if record already exists
228 if (recordExists(record.relay_id, record.record_id, record.timestamp))
229 {
230 return true; // Not an error, just skip duplicate
231 }
232
233 const char* sql = "INSERT INTO ser_records (relay_id, relay_name, record_id, timestamp, status, description) "
234 "VALUES (?, ?, ?, ?, ?, ?);";
235
236 sqlite3_stmt* stmt;
237 int rc = sqlite3_prepare_v2(db_, sql, -1, &stmt, nullptr);
238
239 if (rc != SQLITE_OK)
240 {
241 last_error_ = "Failed to prepare statement: " + std::string(sqlite3_errmsg(db_));
242 return false;
243 }
244
245 sqlite3_bind_text(stmt, 1, record.relay_id.c_str(), -1, SQLITE_TRANSIENT);
246 sqlite3_bind_text(stmt, 2, record.relay_name.c_str(), -1, SQLITE_TRANSIENT);
247 sqlite3_bind_text(stmt, 3, record.record_id.c_str(), -1, SQLITE_TRANSIENT);
248 sqlite3_bind_text(stmt, 4, record.timestamp.c_str(), -1, SQLITE_TRANSIENT);
249 sqlite3_bind_text(stmt, 5, record.status.c_str(), -1, SQLITE_TRANSIENT);
250 sqlite3_bind_text(stmt, 6, record.description.c_str(), -1, SQLITE_TRANSIENT);
251
252 rc = sqlite3_step(stmt);
253 sqlite3_finalize(stmt);
254
255 if (rc != SQLITE_DONE)
256 {
257 last_error_ = "Failed to insert record: " + std::string(sqlite3_errmsg(db_));
258 return false;
259 }
260
261 return true;
262}

266 *
267 * @details Wraps all inserts in `BEGIN TRANSACTION / COMMIT` for
268 * significantly better performance on large batches. Each
269 * record is inserted via insertRecord(), so duplicates are
270 * silently skipped.
271 *
272 * @param records Vector of SERRecord objects to insert.
273 *
274 * @return int Number of records successfully inserted (excludes duplicates).
275 * Returns 0 if the database is not open.
276 *
277 * @pre isOpen() == true
278 *
279 * @note The entire batch is committed even if some individual inserts
280 * are skipped as duplicates.
281 */
282int SERDatabase::insertRecords(const std::vector<SERRecord>& records)
283{
284 if (!db_)
285 {
286 last_error_ = "Database not open";
287 return 0;
288 }
289
290 int inserted = 0;
291
292 // Begin transaction for better performance
293 sqlite3_exec(db_, "BEGIN TRANSACTION;", nullptr, nullptr, nullptr);
294
295 for (const auto& record : records)
296 {
297 if (insertRecord(record))
298 {
299 inserted++;
300 }
301 }
302
303 sqlite3_exec(db_, "COMMIT;", nullptr, nullptr, nullptr);
304
305 return inserted;
306}

323std::vector<SERRecord> SERDatabase::insertAndGetNewRecords(
324 const std::vector<SERRecord>& records, int& insertedCount)
325{
326 insertedCount = 0;
327 std::vector<SERRecord> newRecords;
328
329 if (!db_)
330 {
331 last_error_ = "Database not open";
332 return newRecords;
333 }
334
335 sqlite3_exec(db_, "BEGIN TRANSACTION;", nullptr, nullptr, nullptr);
336
337 for (const auto& record : records)
338 {
339 // Check if already exists BEFORE inserting
340 if (recordExists(record.relay_id, record.record_id, record.timestamp))
341 continue; // duplicate — skip
342
343 if (insertRecord(record))
344 {
345 ++insertedCount;
346 newRecords.push_back(record);
347 }
348 }
349
350 sqlite3_exec(db_, "COMMIT;", nullptr, nullptr, nullptr);
351
352 return newRecords;
353}

371std::vector<SERRecord> SERDatabase::getAllRecords()
372{
373 std::vector<SERRecord> records;
374
375 if (!db_)
376 {
377 last_error_ = "Database not open";
378 return records;
379 }
380
381 const char* sql = "SELECT relay_id, relay_name, record_id, timestamp, status, description FROM ser_records "
382 "ORDER BY timestamp DESC;";
383
384 sqlite3_stmt* stmt;
385 int rc = sqlite3_prepare_v2(db_, sql, -1, &stmt, nullptr);
386
387 if (rc != SQLITE_OK)
388 {
389 last_error_ = "Failed to prepare statement: " + std::string(sqlite3_errmsg(db_));
390 return records;
391 }
392
393 while (sqlite3_step(stmt) == SQLITE_ROW)
394 {
395 SERRecord record;
396 record.relay_id = reinterpret_cast<const char*>(sqlite3_column_text(stmt, 0));
397 record.relay_name = reinterpret_cast<const char*>(sqlite3_column_text(stmt, 1));
398 record.record_id = reinterpret_cast<const char*>(sqlite3_column_text(stmt, 2));
399 record.timestamp = reinterpret_cast<const char*>(sqlite3_column_text(stmt, 3));
400 record.status = reinterpret_cast<const char*>(sqlite3_column_text(stmt, 4));
401
402 const char* desc = reinterpret_cast<const char*>(sqlite3_column_text(stmt, 5));
403 record.description = desc ? desc : "";
404
405 records.push_back(record);
406 }
407
408 sqlite3_finalize(stmt);
409 return records;

611bool SERDatabase::recordExists(const std::string& relayId, const std::string& recordId, const std::string& timestamp)
612{
613 const char* sql = "SELECT 1 FROM ser_records WHERE relay_id = ? AND record_id = ? AND timestamp = ? LIMIT 1;";

620 return false;

633// ================= EXPORT TO JSON =================
634
640 * `sno`, `date`, `time`, `element`, and `state` fields.
641 *
642 * Special characters in the `description` field are escaped
643 * to produce valid JSON output.
644 *
645 * @param filePath Destination file path (e.g. "data.json"). Created or
646 * overwritten if it already exists.
647 *
648 * @return true File written successfully.
649 * @return false Database not open or file cannot be opened for writing.
650 * Check getLastError() for details.
651 *
652 * @pre isOpen() == true
653 *
654 * @see SERJsonWriter Alternative streaming JSON writer.
655 */
656bool SERDatabase::exportToJSON(const std::string& filePath)
657{
658 if (!db_)
659 {
660 last_error_ = "Database not open";
661 return false;
662 }
663
664 std::vector<SERRecord> records = getAllRecords();
665
666 std::ofstream file(filePath);
667 if (!file.is_open())
668 {
669 last_error_ = "Cannot open file for writing: " + filePath;
670 return false;
671 }
672
673 file << "[\n";
674
675 int sno = 1;
676 for (size_t i = 0; i < records.size(); ++i)
677 {
678 const auto& rec = records[i];
679
680 // Parse timestamp to extract date and time
681 std::string date = rec.timestamp;
682 std::string time = "";
683
684 size_t spacePos = rec.timestamp.find(' ');
685 if (spacePos != std::string::npos)
686 {
687 date = rec.timestamp.substr(0, spacePos);
688 time = rec.timestamp.substr(spacePos + 1);
689 }
690
691 // Escape special characters in description
692 std::string element = rec.description;
693 for (size_t j = 0; j < element.length(); ++j)
694 {
695 if (element[j] == '"' || element[j] == '\\')
696 {
697 element.insert(j, "\\");
698 ++j;
699 }
700 }
701
702 file << " {\n";
703 file << " \"sno\": " << sno++ << ",\n";
704 file << " \"relay_id\": \"" << rec.relay_id << "\",\n";
705 file << " \"relay_name\": \"" << rec.relay_name << "\",\n";
706 file << " \"date\": \"" << date << "\",\n";
707 file << " \"time\": \"" << time << "\",\n";
708 file << " \"element\": \"" << element << "\",\n";
709 file << " \"state\": \"" << rec.status << "\"\n";
710 file << " }";
711
712 if (i < records.size() - 1)
713 file << ",";
714
715 file << "\n";
716 }
717
718 file << "]\n";
719 file.close();
720
721 return true;
722}
