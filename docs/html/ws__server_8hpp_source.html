159inline std::string recordsToJSON(const std::vector<SERRecord>& records)
160{
161 std::string json = "[\n";
162
163 for (size_t i = 0; i < records.size(); ++i)
164 {
165 const auto& rec = records[i];
166
167 // Parse timestamp to extract date and time
168 std::string date = rec.timestamp;
169 std::string time = "";
170
171 size_t spacePos = rec.timestamp.find(' ');
172 if (spacePos != std::string::npos)
173 {
174 date = rec.timestamp.substr(0, spacePos);
175 time = rec.timestamp.substr(spacePos + 1);
176 }
177
178 // Escape special characters in description
179 std::string element = rec.description;
180 for (size_t j = 0; j < element.length(); ++j)
181 {
182 if (element[j] == '"' || element[j] == '\\')
183 {
184 element.insert(j, "\\");
185 ++j;
186 }
187 }
188
189 json += " {\n";
190 json += " \"sno\": " + rec.record_id + ",\n"; // Use original # from relay
191 json += " \"date\": \"" + date + "\",\n";
192 json += " \"time\": \"" + time + "\",\n";
193 json += " \"element\": \"" + element + "\",\n";
194 json += " \"state\": \"" + rec.status + "\"\n";
195 json += " }";
196
197 if (i < records.size() - 1)
198 json += ",";
199
200 json += "\n";
201 }
202
203 json += "]";
204 return json;
205}

520 void drainQueue()
521 {
522 if (writing_ || write_queue_.empty())
523 return;
524
525 auto msg = std::move(write_queue_.front());
526 write_queue_.pop_front();
527
528 switch (msg.kind)
529 {
530 case PendingMsg::Binary:
531 writing_ = true;
532 broadcast_buffer_ = std::move(msg.binary_data);
533 ws_.binary(true);
534 ws_.async_write(
535 net::buffer(broadcast_buffer_),
536 beast::bind_front_handler(&WebSocketSession::on_broadcast_write,
537 shared_from_this()));
538 break;
539
540 case PendingMsg::Text:
541 writing_ = true;
542 text_write_buffer_ = std::move(msg.text_data);
543 ws_.text(true);
544 ws_.async_write(
545 net::buffer(text_write_buffer_),
546 beast::bind_front_handler(&WebSocketSession::on_broadcast_write,
547 shared_from_this()));
548 break;
549
550 case PendingMsg::SendAll:
551 sendData(); // will set writing_ itself
552 break;
553 }
554 }

572 void on_read(beast::error_code ec, std::size_t bytes_transferred)
573 {
574 boost::ignore_unused(bytes_transferred);
575
576 if (ec == websocket::error::closed)
577 {
578 std::cout << "[WS] Client disconnected\n";
579 // Deregister from session manager
580 if (sessionMgr_)
581 sessionMgr_->remove(shared_from_this());
582 return;
583 }
584
585 if (ec)
586 {
587 std::cerr << "[WS] Read error: " << ec.message() << "\n";
588 // Deregister from session manager on error too
589 if (sessionMgr_)
590 sessionMgr_->remove(shared_from_this());
591 return;
592 }
593
594 // Handle incoming message
595 std::string msg = beast::buffers_to_string(buffer_.data());
596 buffer_.consume(buffer_.size());
597
598 // Mask sensitive payloads (passwords) in log output
599 if (isJsonAction(msg) && extractAction(msg) == "change_password")
600 std::cout << "[WS] Received: {\"action\":\"change_password\",...}\n";
601 else
602 std::cout << "[WS] Received: " << msg << "\n";
603
604 if (msg == "refresh" || msg == "getData")
605 {
606 // Send data, then continue reading when write completes
607 pending_read_ = true;
608 sendData();
609 }
610 else if (actionHandler_ && isJsonAction(msg))
611 {
612 // JSON action (e.g. read_time, sync_time, change_password) — handle asynchronously
613 std::string action = extractAction(msg);
614 if (action == "change_password")
615 std::cout << "[WS] JSON action detected: " << action << " (payload masked)\n";
616 else
617 std::cout << "[WS] JSON action detected: " << action << "\n";
618
619 // Run handler on a detached thread to avoid blocking the strand
620 // Pass full JSON message so handlers can extract extra fields
621 auto self = shared_from_this();
622 auto handler = actionHandler_;
623 auto fullMsg = msg;
624 std::thread([self, handler, fullMsg]() {
625 std::string response = handler(fullMsg);
626 // Post text response back onto the session's strand
627 net::post(self->ws_.get_executor(), [self, response]() {
628 self->sendTextResponse(response);
629 self->do_read();
630 });
631 }).detach();
632 }
633 else if (streamCmdHandler_ || cmdHandler_)
634 {
635 // Forward command to relay — try streaming handler first, fallback to single-response
636 std::cout << "[WS] Forwarding command via handler: " << msg << "\n";
637 auto self = shared_from_this();
638 auto streamHandler = streamCmdHandler_;
639 auto handler = cmdHandler_;
640 auto cmdMsg = msg;
641 streaming_abort_.store(false);
642 std::thread([self, streamHandler, handler, cmdMsg]() {
643 bool streamed = false;
644 if (streamHandler)
645 {
646 streamed = streamHandler(cmdMsg,
647 [self](const std::string& text) {
648 net::post(self->ws_.get_executor(), [self, text]() {
649 self->sendTextResponse(text);
650 });
651 },
652 self->streaming_abort_
653 );
654 }
655 if (!streamed && handler)
656 {
657 std::string response = handler(cmdMsg);
658 net::post(self->ws_.get_executor(), [self, response]() {
659 if (!response.empty())
660 self->sendTextResponse(response);
661 });
662 }
663 net::post(self->ws_.get_executor(), [self]() {
664 self->do_read();
665 });
666 }).detach();
667 }
668 else
669 {
670 // No handler registered, ignore
671 do_read();
672 }
673 }

681 void sendData()
682 {
683 if (writing_)
684 {
685 // Queue a full-DB send for when the current write finishes
686 write_queue_.push_back({PendingMsg::SendAll, {}, {}});
687 return;
688 }
689
690 writing_ = true;
691
692 auto records = db_.getAllRecords();
693 write_buffer_ = asn_tlv::encodeSerRecordsToTlv(records); // Store in member to keep alive
694
695 ws_.binary(true);
696 ws_.async_write(
697 net::buffer(write_buffer_), // Use member buffer
698 beast::bind_front_handler(&WebSocketSession::on_write, shared_from_this()));
699 }

707 void on_write(beast::error_code ec, std::size_t bytes_transferred)
708 {

710
711 writing_ = false;
712
713 if (ec)
714 {
715 std::cerr << "[WS] Write error: " << ec.message() << "\n";
716 streaming_abort_.store(true);
717 write_queue_.clear();
718 return;
719 }
720
721 // Start reading if a read was pending
722 if (pending_read_)
723 {
724 pending_read_ = false;
725 do_read();
726 }
727
728 // Flush any queued writes (broadcasts / text / sendAll)
729 drainQueue();
730 }

735 static bool isJsonAction(const std::string& msg)
736 {
737 // Quick heuristic: starts with '{' and contains "action"
738 if (msg.empty() || msg[0] != '{') return false;
739 return msg.find("\"action\"") != std::string::npos;
740 }
741
747 static std::string extractAction(const std::string& json)
748 {
749 const std::string key = "\"action\"";
750 auto pos = json.find(key);
751 if (pos == std::string::npos) return {};
752 pos = json.find(':', pos + key.size());
753 if (pos == std::string::npos) return {};
754 pos = json.find('"', pos + 1);
755 if (pos == std::string::npos) return {};
756 auto end = json.find('"', pos + 1);
757 if (end == std::string::npos) return {};
758 return json.substr(pos + 1, end - pos - 1);
759 }

821public:
833 WebSocketListener(net::io_context& ioc, tcp::endpoint endpoint, SERDatabase& db,

836 WebSocketSession::ActionHandler actionHandler = nullptr,
837 WebSocketSession::StreamCommandHandler streamCmdHandler = nullptr)

849 acceptor_.open(endpoint.protocol(), ec);
850 if (ec)

852 std::cerr << "[WS] Open error: " << ec.message() << "\n";
853 return;
854 }
855
856 // Allow address reuse
857 acceptor_.set_option(net::socket_base::reuse_address(true), ec);
858 if (ec)
859 {
860 std::cerr << "[WS] Set option error: " << ec.message() << "\n";
861 return;
862 }
863
864 // Bind to the server address
865 acceptor_.bind(endpoint, ec);
866 if (ec)
867 {
868 std::cerr << "[WS] Bind error: " << ec.message() << "\n";
869 return;
870 }
871
872 // Start listening for connections
873 acceptor_.listen(net::socket_base::max_listen_connections, ec);
874 if (ec)
875 {
876 std::cerr << "[WS] Listen error: " << ec.message() << "\n";
877 return;
878 }
879 }

887 {
888 do_accept();
889 }
890
891private:

897 acceptor_.async_accept(
898 net::make_strand(ioc_),
