157inline std::string recordsToJSON(const std::vector<SERRecord>& records)
158{
159 std::string json = "[\n";
160
161 for (size_t i = 0; i < records.size(); ++i)
162 {
163 const auto& rec = records[i];
164
165 // Parse timestamp to extract date and time
166 std::string date = rec.timestamp;
167 std::string time = "";
168
169 size_t spacePos = rec.timestamp.find(' ');
170 if (spacePos != std::string::npos)
171 {
172 date = rec.timestamp.substr(0, spacePos);
173 time = rec.timestamp.substr(spacePos + 1);
174 }
175
176 // Escape special characters in description
177 std::string element = rec.description;
178 for (size_t j = 0; j < element.length(); ++j)
179 {
180 if (element[j] == '"' || element[j] == '\\')
181 {
182 element.insert(j, "\\");
183 ++j;
184 }
185 }
186
187 json += " {\n";
188 json += " \"sno\": " + rec.record_id + ",\n"; // Use original # from relay
189 json += " \"date\": \"" + date + "\",\n";
190 json += " \"time\": \"" + time + "\",\n";
191 json += " \"element\": \"" + element + "\",\n";
192 json += " \"state\": \"" + rec.status + "\"\n";
193 json += " }";
194
195 if (i < records.size() - 1)
196 json += ",";
197
198 json += "\n";
199 }
200
201 json += "]";
202 return json;
203}

499 void on_read(beast::error_code ec, std::size_t bytes_transferred)
500 {
501 boost::ignore_unused(bytes_transferred);
502
503 if (ec == websocket::error::closed)
504 {
505 std::cout << "[WS] Client disconnected\n";
506 // Deregister from session manager
507 if (sessionMgr_)
508 sessionMgr_->remove(shared_from_this());
509 return;
510 }
511
512 if (ec)
513 {
514 std::cerr << "[WS] Read error: " << ec.message() << "\n";
515 // Deregister from session manager on error too
516 if (sessionMgr_)
517 sessionMgr_->remove(shared_from_this());
518 return;
519 }
520
521 // Handle incoming message
522 std::string msg = beast::buffers_to_string(buffer_.data());
523 buffer_.consume(buffer_.size());
524
525 // Mask sensitive payloads (passwords) in log output
526 if (isJsonAction(msg) && extractAction(msg) == "change_password")
527 std::cout << "[WS] Received: {\"action\":\"change_password\",...}\n";
528 else
529 std::cout << "[WS] Received: " << msg << "\n";
530
531 if (msg == "refresh" || msg == "getData")
532 {
533 // Send data, then continue reading when write completes
534 pending_read_ = true;
535 sendData();
536 }
537 else if (actionHandler_ && isJsonAction(msg))
538 {
539 // JSON action (e.g. read_time, sync_time, change_password) — handle asynchronously
540 std::string action = extractAction(msg);
541 if (action == "change_password")
542 std::cout << "[WS] JSON action detected: " << action << " (payload masked)\n";
543 else
544 std::cout << "[WS] JSON action detected: " << action << "\n";
545
546 // Run handler on a detached thread to avoid blocking the strand
547 // Pass full JSON message so handlers can extract extra fields
548 auto self = shared_from_this();
549 auto handler = actionHandler_;
550 auto fullMsg = msg;
551 std::thread([self, handler, fullMsg]() {
552 std::string response = handler(fullMsg);
553 // Post text response back onto the session's strand
554 net::post(self->ws_.get_executor(), [self, response]() {
555 self->sendTextResponse(response);
556 self->do_read();
557 });
558 }).detach();
559 }
560 else if (cmdHandler_)
561 {
562 // Queue command to ReceptionWorker (async — response comes via broadcastText)
563 std::cout << "[WS] Queuing command via handler: " << msg << "\n";
564 cmdHandler_(msg);
565 do_read();
566 }
567 else
568 {
569 // No handler registered, ignore
570 do_read();
571 }
572 }
573
580 void sendData()
581 {
582 // Prevent concurrent writes
583 if (writing_)
584 {
585 std::cout << "[WS] Write already in progress, skipping\n";
586 return;
587 }
588
589 writing_ = true;
590
591 auto records = db_.getAllRecords();
592 write_buffer_ = asn_tlv::encodeSerRecordsToTlv(records); // Store in member to keep alive
593
594 ws_.binary(true);
595 ws_.async_write(
596 net::buffer(write_buffer_), // Use member buffer
597 beast::bind_front_handler(&WebSocketSession::on_write, shared_from_this()));
598 }
599
606 void on_write(beast::error_code ec, std::size_t bytes_transferred)
607 {
608 boost::ignore_unused(bytes_transferred);
609
610 writing_ = false;
611
612 if (ec)
613 {
614 std::cerr << "[WS] Write error: " << ec.message() << "\n";
615 return;
616 }
617
618 // Start reading if a read was pending
619 if (pending_read_)
620 {
621 pending_read_ = false;
622 do_read();
623 }
624 }

629 static bool isJsonAction(const std::string& msg)
630 {
631 // Quick heuristic: starts with '{' and contains "action"
632 if (msg.empty() || msg[0] != '{') return false;
633 return msg.find("\"action\"") != std::string::npos;
634 }
635
641 static std::string extractAction(const std::string& json)
642 {
643 const std::string key = "\"action\"";
644 auto pos = json.find(key);
645 if (pos == std::string::npos) return {};
646 pos = json.find(':', pos + key.size());
647 if (pos == std::string::npos) return {};
648 pos = json.find('"', pos + 1);
649 if (pos == std::string::npos) return {};
650 auto end = json.find('"', pos + 1);
651 if (end == std::string::npos) return {};
652 return json.substr(pos + 1, end - pos - 1);
653 }
654
660 void sendTextResponse(const std::string& text)
661 {
662 if (writing_)
663 {
664 std::cout << "[WS] Write in progress, skipping text response\n";
665 return;
666 }
667
668 writing_ = true;
669 text_write_buffer_ = text;
670
671 ws_.text(true); // Send as text frame (not binary)
672 ws_.async_write(
673 net::buffer(text_write_buffer_),
674 beast::bind_front_handler(&WebSocketSession::on_write, shared_from_this()));
675 }

704class TELNET_SML_API WebSocketListener : public std::enable_shared_from_this<WebSocketListener>
705{
706 net::io_context& ioc_;
707 tcp::acceptor acceptor_;
708 SERDatabase& db_;
709 SessionManager* sessionMgr_;
710 WebSocketSession::CommandHandler cmdHandler_;
711 WebSocketSession::ActionHandler actionHandler_;
712
713public:
724 WebSocketListener(net::io_context& ioc, tcp::endpoint endpoint, SERDatabase& db,
725 SessionManager* sessionMgr = nullptr,
726 WebSocketSession::CommandHandler cmdHandler = nullptr,
727 WebSocketSession::ActionHandler actionHandler = nullptr)
728 : ioc_(ioc)
729 , acceptor_(ioc)
730 , db_(db)
731 , sessionMgr_(sessionMgr)
732 , cmdHandler_(std::move(cmdHandler))
733 , actionHandler_(std::move(actionHandler))
734 {
735 beast::error_code ec;
736
737 // Open the acceptor
738 acceptor_.open(endpoint.protocol(), ec);
739 if (ec)
740 {
741 std::cerr << "[WS] Open error: " << ec.message() << "\n";
742 return;
743 }
744
745 // Allow address reuse
746 acceptor_.set_option(net::socket_base::reuse_address(true), ec);
747 if (ec)
748 {
749 std::cerr << "[WS] Set option error: " << ec.message() << "\n";
750 return;
751 }
752
753 // Bind to the server address
754 acceptor_.bind(endpoint, ec);
755 if (ec)
756 {
757 std::cerr << "[WS] Bind error: " << ec.message() << "\n";
758 return;
759 }
760
761 // Start listening for connections
762 acceptor_.listen(net::socket_base::max_listen_connections, ec);
763 if (ec)
764 {
765 std::cerr << "[WS] Listen error: " << ec.message() << "\n";
766 return;
767 }
768 }
769
775 void run()
776 {
777 do_accept();
778 }
779
780private:
784 void do_accept()
785 {
786 acceptor_.async_accept(
787 net::make_strand(ioc_),
788 beast::bind_front_handler(&WebSocketListener::on_accept, shared_from_this()));
789 }
790
797 void on_accept(beast::error_code ec, tcp::socket socket)
798 {
799 if (ec)
800 {
801 std::cerr << "[WS] Accept error: " << ec.message() << "\n";
802 }
803 else
804 {
805 // Create the session and run it (with session manager + command handler + action handler)
806 std::make_shared<WebSocketSession>(std::move(socket), db_, sessionMgr_, cmdHandler_, actionHandler_)->run();
807 }
808
809 // Accept another connection
810 do_accept();
811 }
