140inline std::string recordsToJSON(const std::vector<SERRecord>& records)
141{
142 std::string json = "[\n";
143
144 for (size_t i = 0; i < records.size(); ++i)
145 {
146 const auto& rec = records[i];
147
148 // Parse timestamp to extract date and time
149 std::string date = rec.timestamp;
150 std::string time = "";
151
152 size_t spacePos = rec.timestamp.find(' ');
153 if (spacePos != std::string::npos)
154 {
155 date = rec.timestamp.substr(0, spacePos);
156 time = rec.timestamp.substr(spacePos + 1);
157 }
158
159 // Escape special characters in description
160 std::string element = rec.description;
161 for (size_t j = 0; j < element.length(); ++j)
162 {
163 if (element[j] == '"' || element[j] == '\\')
164 {
165 element.insert(j, "\\");
166 ++j;
167 }
168 }
169
170 json += " {\n";
171 json += " \"sno\": " + rec.record_id + ",\n"; // Use original # from relay
172 json += " \"date\": \"" + date + "\",\n";
173 json += " \"time\": \"" + time + "\",\n";
174 json += " \"element\": \"" + element + "\",\n";
175 json += " \"state\": \"" + rec.status + "\"\n";
176 json += " }";
177
178 if (i < records.size() - 1)
179 json += ",";
180
181 json += "\n";
182 }
183
184 json += "]";
185 return json;
186}
