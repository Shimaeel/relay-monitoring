157inline std::string recordsToJSON(const std::vector<SERRecord>& records)
158{
159 std::string json = "[\n";
160
161 for (size_t i = 0; i < records.size(); ++i)
162 {
163 const auto& rec = records[i];
164
165 // Parse timestamp to extract date and time
166 std::string date = rec.timestamp;
167 std::string time = "";
168
169 size_t spacePos = rec.timestamp.find(' ');
170 if (spacePos != std::string::npos)
171 {
172 date = rec.timestamp.substr(0, spacePos);
173 time = rec.timestamp.substr(spacePos + 1);
174 }
175
176 // Escape special characters in description
177 std::string element = rec.description;
178 for (size_t j = 0; j < element.length(); ++j)
179 {
180 if (element[j] == '"' || element[j] == '\\')
181 {
182 element.insert(j, "\\");
183 ++j;
184 }
185 }
186
187 json += " {\n";
188 json += " \"sno\": " + rec.record_id + ",\n"; // Use original # from relay
189 json += " \"date\": \"" + date + "\",\n";
190 json += " \"time\": \"" + time + "\",\n";
191 json += " \"element\": \"" + element + "\",\n";
192 json += " \"state\": \"" + rec.status + "\"\n";
193 json += " }";
194
195 if (i < records.size() - 1)
196 json += ",";
197
198 json += "\n";
199 }
200
201 json += "]";
202 return json;
203}

323 std::vector<uint8_t> broadcast_buffer_;
324
325public:
333 explicit WebSocketSession(tcp::socket&& socket, SERDatabase& db, SessionManager* sessionMgr = nullptr)
334 : ws_(std::move(socket))
335 , db_(db)
336 , sessionMgr_(sessionMgr)
337 {
338 }
339
343 ~WebSocketSession()
344 {
345 // Note: Cannot use shared_from_this() in destructor
346 // Deregistration is handled in on_read when connection closes
347 }
348
357 void sendBroadcast(const std::vector<uint8_t>& data)
358 {
359 // Post to strand to ensure thread safety
360 net::post(ws_.get_executor(), [self = shared_from_this(), data]() {
361 self->do_broadcast(data);
362 });
363 }
364
371 void run()
372 {
373 // Set suggested timeout settings for the websocket
374 ws_.set_option(websocket::stream_base::timeout::suggested(beast::role_type::server));
375
376 // Set a decorator to change the Server of the handshake
377 ws_.set_option(websocket::stream_base::decorator(
378 [](websocket::response_type& res) {
379 res.set(http::field::server, "SER-WebSocket-Server");
380 }));
381
382 // Accept the websocket handshake
383 ws_.async_accept(
384 beast::bind_front_handler(&WebSocketSession::on_accept, shared_from_this()));
385 }
386
387private:
393 void on_accept(beast::error_code ec)
394 {
395 if (ec)
396 {
397 std::cerr << "[WS] Accept error: " << ec.message() << "\n";
398 return;
399 }
400
401 std::cout << "[WS] Client connected\n";
402
403 // Register with session manager for broadcast
404 if (sessionMgr_)
405 sessionMgr_->add(shared_from_this());
406
407 // Send initial data, then start reading when write completes
408 pending_read_ = true;
409 sendData();
410 }
411
415 void do_broadcast(const std::vector<uint8_t>& data)
416 {
417 if (writing_)
418 {
419 // Queue for later or skip - for simplicity, skip if busy
420 return;
421 }
422
423 writing_ = true;
424 broadcast_buffer_ = data;
425
426 ws_.binary(true);
427 ws_.async_write(
428 net::buffer(broadcast_buffer_),
429 beast::bind_front_handler(&WebSocketSession::on_broadcast_write, shared_from_this()));
430 }
431
435 void on_broadcast_write(beast::error_code ec, std::size_t /*bytes_transferred*/)
436 {
437 writing_ = false;
438
439 if (ec)
440 {
441 std::cerr << "[WS] Broadcast write error: " << ec.message() << "\n";
442 }
443 }
444
448 void do_read()
449 {
450 ws_.async_read(
451 buffer_,
452 beast::bind_front_handler(&WebSocketSession::on_read, shared_from_this()));
453 }
454
461 void on_read(beast::error_code ec, std::size_t bytes_transferred)
462 {
463 boost::ignore_unused(bytes_transferred);
464
465 if (ec == websocket::error::closed)
466 {
467 std::cout << "[WS] Client disconnected\n";
468 // Deregister from session manager
469 if (sessionMgr_)
470 sessionMgr_->remove(shared_from_this());
471 return;
472 }
473
474 if (ec)
475 {
476 std::cerr << "[WS] Read error: " << ec.message() << "\n";
477 // Deregister from session manager on error too
478 if (sessionMgr_)
479 sessionMgr_->remove(shared_from_this());
480 return;
481 }
482
483 // Handle incoming message
484 std::string msg = beast::buffers_to_string(buffer_.data());
485 buffer_.consume(buffer_.size());
486
487 std::cout << "[WS] Received: " << msg << "\n";
488
489 if (msg == "refresh" || msg == "getData")
490 {
491 // Send data, then continue reading when write completes
492 pending_read_ = true;
493 sendData();
494 }
495 else
496 {
497 // Continue reading for other messages
498 do_read();
499 }
500 }
501
508 void sendData()
509 {
510 // Prevent concurrent writes
511 if (writing_)
512 {
513 std::cout << "[WS] Write already in progress, skipping\n";
514 return;
515 }
516
517 writing_ = true;
518
519 auto records = db_.getAllRecords();
520 write_buffer_ = asn_tlv::encodeSerRecordsToTlv(records); // Store in member to keep alive
521
522 ws_.binary(true);
523 ws_.async_write(
524 net::buffer(write_buffer_), // Use member buffer
525 beast::bind_front_handler(&WebSocketSession::on_write, shared_from_this()));
526 }
527
534 void on_write(beast::error_code ec, std::size_t bytes_transferred)
535 {
536 boost::ignore_unused(bytes_transferred);
537
538 writing_ = false;
539
540 if (ec)
541 {
542 std::cerr << "[WS] Write error: " << ec.message() << "\n";
543 return;
544 }
545
546 // Start reading if a read was pending
547 if (pending_read_)
548 {
549 pending_read_ = false;
550 do_read();
551 }
552 }
553};

581class TELNET_SML_API WebSocketListener : public std::enable_shared_from_this<WebSocketListener>
582{
583 net::io_context& ioc_;
584 tcp::acceptor acceptor_;
585 SERDatabase& db_;
586 SessionManager* sessionMgr_;
587
588public:
597 WebSocketListener(net::io_context& ioc, tcp::endpoint endpoint, SERDatabase& db, SessionManager* sessionMgr = nullptr)
598 : ioc_(ioc)
599 , acceptor_(ioc)
600 , db_(db)
601 , sessionMgr_(sessionMgr)
602 {
603 beast::error_code ec;
604
605 // Open the acceptor
606 acceptor_.open(endpoint.protocol(), ec);
607 if (ec)
608 {
609 std::cerr << "[WS] Open error: " << ec.message() << "\n";
610 return;
611 }
612
613 // Allow address reuse
614 acceptor_.set_option(net::socket_base::reuse_address(true), ec);
615 if (ec)
616 {
617 std::cerr << "[WS] Set option error: " << ec.message() << "\n";
618 return;
619 }
620
621 // Bind to the server address
622 acceptor_.bind(endpoint, ec);
623 if (ec)
624 {
625 std::cerr << "[WS] Bind error: " << ec.message() << "\n";
626 return;
627 }
628
629 // Start listening for connections
630 acceptor_.listen(net::socket_base::max_listen_connections, ec);
631 if (ec)
632 {
633 std::cerr << "[WS] Listen error: " << ec.message() << "\n";
634 return;
635 }
636 }
637
643 void run()
644 {
645 do_accept();
646 }
647
648private:
652 void do_accept()
653 {
654 acceptor_.async_accept(
655 net::make_strand(ioc_),
656 beast::bind_front_handler(&WebSocketListener::on_accept, shared_from_this()));
657 }
658
665 void on_accept(beast::error_code ec, tcp::socket socket)
666 {
667 if (ec)
668 {
669 std::cerr << "[WS] Accept error: " << ec.message() << "\n";
670 }
671 else
672 {
673 // Create the session and run it (with session manager for broadcast)
674 std::make_shared<WebSocketSession>(std::move(socket), db_, sessionMgr_)->run();
675 }
676
677 // Accept another connection
678 do_accept();
679 }

789 return false;
790 }
791 }

800 void stop()
801 {
802 if (!running_)
803 return;
804
805 ioc_.stop();
806 if (server_thread_.joinable())
807 server_thread_.join();
808
809 running_ = false;
810 std::cout << "[WS] Server stopped\n";
811 }

819 bool isRunning() const { return running_; }

829 void broadcast(const std::vector<SERRecord>& records)
830 {
831 if (!running_)

839 std::cout << "[WS] Broadcasting to " << sessions.size() << " clients\n";
840

844 session->sendBroadcast(payload);
