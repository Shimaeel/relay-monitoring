209 bool push(RawDataMessage msg)
210 {
211 bool dropped = false;
212 {
213 std::lock_guard<std::mutex> lock(mutex_);
214
215 // Drop oldest if at capacity (never block producer)
216 while (queue_.size() >= max_size_)
217 {
218 // Advance any reader still pointing at the oldest message
219 for (auto& r : readers_)
220 {
221 if (r.active && r.read_index <= base_index_)
222 r.read_index = base_index_ + 1U;
223 }
224 queue_.pop_front();
225 ++base_index_;
226 dropped = true;
227 ++drop_count_;
228 }
229
230 queue_.push_back(std::move(msg));
231 ++total_writes_;
232 }
233 cv_.notify_all(); // wake ALL waiting consumers
234 return !dropped;
235 }

251 std::optional<RawDataMessage> waitPop(ReaderId readerId, std::atomic<bool>& stopFlag)
252 {
253 if (readerId >= kMaxReaders)
254 return std::nullopt;
255
256 std::unique_lock<std::mutex> lock(mutex_);
257
258 cv_.wait(lock, [&]() {
259 return stopFlag.load(std::memory_order_relaxed)
260 || (readers_[readerId].active
261 && readers_[readerId].read_index < base_index_ + queue_.size());
262 });
263
264 if (stopFlag.load(std::memory_order_relaxed))
265 return std::nullopt;
266
267 auto& reader = readers_[readerId];
268 if (!reader.active || reader.read_index >= base_index_ + queue_.size())
269 return std::nullopt;
270
271 std::size_t dequeIdx = reader.read_index - base_index_;
272 RawDataMessage msg = queue_[dequeIdx]; // COPY — other readers still need it
273 ++reader.read_index;
274 ++reader.total_reads;
275
276 trimConsumed();
277 return msg;
278 }

426 void trimConsumed()
427 {
428 std::size_t minIdx = base_index_ + queue_.size();
429 bool anyActive = false;
430
431 for (const auto& r : readers_)
432 {
433 if (r.active)
434 {
435 anyActive = true;
436 if (r.read_index < minIdx)
437 minIdx = r.read_index;
438 }
439 }
440
441 if (!anyActive)
442 return; // No active readers — keep messages for future readers
443
444 // Remove messages before the slowest reader
445 while (base_index_ < minIdx && !queue_.empty())
446 {
447 queue_.pop_front();
448 ++base_index_;
449 }
450 }
