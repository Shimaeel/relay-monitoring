113 // ─── FSM Driver ─────────────────────────────────────────────────────────
114
125 bool driveToOperational()
126 {
127 using namespace sml;
128
129 while (!stop_flag_.load())
130 {
131 if (fsm_.is("Operational"_s))
132 return true;
133
134 if (fsm_.is("Error"_s))
135 {
136 std::cout << relay_tag_ << " FSM reached Error state\n";
137 return false;
138 }
139
140 // Interruptible wait for retry states
141 if (fsm_.is("ConnectWait"_s) || fsm_.is("LoginRetryWait"_s))
142 {
143 std::cout << relay_tag_ << " Waiting " << RETRY_DELAY_SEC
144 << "s before retry...\n";
145 for (int s = 0; s < RETRY_DELAY_SEC && !stop_flag_.load(); ++s)
146 std::this_thread::sleep_for(std::chrono::seconds(1));
147 if (stop_flag_.load())
148 return false;
149 }
150
151 fsm_.process_event(step_event{});
152 }
153 return false;
154 }
155
156 // ─── Command FSM Event Dispatch ──────────────────────────────────────────
157
171 void fireCommandEvent(const std::string& cmd)
172 {
173 cmdResponse_.success = false;
174 cmdResponse_.response.clear();
175
176 if (cmd == "SER")
177 cmdFsm_.process_event(cmd_ser_event{});
178 else if (cmd.size() >= 3 && cmd.substr(0, 3) == "TAR")
179 cmdFsm_.process_event(cmd_tar_event{cmd.size() > 4 ? cmd.substr(4) : ""});
180 else if (cmd == "EVE")
181 cmdFsm_.process_event(cmd_eve_event{});
182 else if (cmd == "CTRL+C")
183 cmdFsm_.process_event(cmd_ctrlc_event{});
184 else if (cmd == "CTRL+D")
185 cmdFsm_.process_event(cmd_ctrld_event{});
186 else if (cmd.size() >= 3 && cmd.substr(0, 3) == "SET")
187 cmdFsm_.process_event(cmd_set_event{cmd.size() > 4 ? cmd.substr(4) : ""});
188 else
189 {
190 // Generic / unknown command — send directly via SER-style action
191 client_.clearLastResponse();
192 std::string response;
193 bool ok = client_.SendCmdReceiveData(cmd, response);
194 cmdResponse_.success = ok && !response.empty();
195 cmdResponse_.response = std::move(response);
196 std::cout << "[CmdFSM] Generic '" << cmd << "' "
197 << (cmdResponse_.success ? "OK" : "FAIL") << "\n";
198 }
199 }
200
201 // ─── Command Execution ──────────────────────────────────────────────────
202
213 void executeCommand(const std::string& cmd)
214 {
215 for (int attempt = 1; attempt <= MAX_RETRIES; ++attempt)
216 {
217 if (stop_flag_.load())
218 return;
219
220 // Phase 1 — ensure Connection FSM is in Operational state
221 if (!driveToOperational())
222 {
223 std::cout << relay_tag_ << " FSM not operational — aborting '"
224 << cmd << "'\n";
225 return;
226 }
227
228 // Phase 2 — send command through Command FSM
229 fireCommandEvent(cmd);
230
231 if (cmdResponse_.success)
232 {
233 rawBuffer_.push({cmd, cmdResponse_.response});
234 std::cout << relay_tag_ << " Pushed " << cmdResponse_.response.size()
235 << " bytes for '" << cmd << "'\n";
236 return;
237 }
238
239 // Phase 3 — command failed → trigger Connection FSM reconnect cycle
240 std::cout << relay_tag_ << " Command '" << cmd << "' failed (attempt "
241 << attempt << "/" << MAX_RETRIES << ")\n";
242 fsm_.process_event(disconnect_event{});
243 }
244
245 std::cout << relay_tag_ << " All retries failed for '" << cmd << "'\n";
246 }
247
248 // ─── Worker Thread ──────────────────────────────────────────────────────
249
250 void runLoop()
251 {
252 std::cout << relay_tag_ << " Worker thread started (FSM-driven)\n";
253
254 // Kick the FSM out of Idle → Connecting (ConnectAction runs)
255 fsm_.process_event(start_event{});
256
257 while (!stop_flag_.load())
258 {
259 std::string cmd;
260 {
261 std::unique_lock<std::mutex> lock(queue_mutex_);
262 queue_cv_.wait(lock, [&]() {
263 return stop_flag_.load() || !command_queue_.empty();
264 });
265
266 if (stop_flag_.load())
267 break;
268
269 if (command_queue_.empty())
270 continue;
271
272 cmd = std::move(command_queue_.front());
273 command_queue_.pop_front();
274 }
275
276 executeCommand(cmd);
277 }
278 std::cout << relay_tag_ << " Worker thread exiting\n";
279 }

345
359 std::string handleUserCommand(const std::string& cmd)
360 {
361 std::lock_guard<std::mutex> lock(sync_cmd_mutex_);
362
363 for (int attempt = 1; attempt <= MAX_RETRIES; ++attempt)
364 {
365 if (stop_flag_.load())
366 return "";
367
368 if (!driveToOperational())
369 return "";
370
371 // Route through Command FSM
372 fireCommandEvent(cmd);
373
374 if (cmdResponse_.success)
375 {
376 std::cout << relay_tag_ << " User cmd '" << cmd
377 << "' OK (" << cmdResponse_.response.size() << " bytes)\n";
378 return cmdResponse_.response;
379 }
380
381 std::cout << relay_tag_ << " User cmd '" << cmd << "' failed (attempt "
382 << attempt << "/" << MAX_RETRIES << ")\n";
383 fsm_.process_event(disconnect_event{});
384 }
385 return "";
386 }
387

396 fsm_.process_event(disconnect_event{});
397 }

419class PipelineProcessingWorker
420{
421 std::atomic<bool> stop_flag_{false};
422 std::thread worker_thread_;
423
424 RawDataRingBuffer& rawBuffer_;
425 SERDatabase& db_;
426 SERWebSocketServer& wsServer_;
427 SharedRingBuffer& shmRing_;
428 std::atomic<bool>& app_running_;
429 RawDataRingBuffer::ReaderId readerId_{RawDataRingBuffer::kInvalidReader};
430
431 std::string relay_id_;
432 std::string relay_name_;
433 std::string relay_tag_;
434
435 void runLoop()
436 {
437 std::cout << relay_tag_ << " Worker thread started\n";
438 while (!stop_flag_.load() && app_running_.load())
439 {
440 auto msg = rawBuffer_.waitPop(readerId_, stop_flag_);
441 if (!msg)
442 continue;
443
444 // Non-SER command — broadcast raw text to all WS clients
445 if (msg->command != "SER")
446 {
447 std::cout << relay_tag_ << " Non-SER command '" << msg->command
448 << "', broadcasting text (" << msg->response.size() << " bytes)\n";
449 wsServer_.broadcastText(msg->response);
450 continue;
451 }
452
453 // Parse SER response
454 auto records = parseSERResponse(msg->response);
455 if (records.empty())
456 {
457 std::cout << relay_tag_ << " No records parsed\n";
458 continue;
459 }
460
461 // *** STAMP relay identity on every record ***
462 for (auto& rec : records)
463 {
464 rec.relay_id = relay_id_;
465 rec.relay_name = relay_name_;
466 }
467
468 std::cout << relay_tag_ << " Parsed " << records.size() << " SER records\n";
469
470 // Step 1: SQLite insert (shared DB, relay_id prevents cross-relay duplicates)
471 int inserted = 0;
472 auto newRecords = db_.insertAndGetNewRecords(records, inserted);
473 std::cout << relay_tag_ << " SQLite: " << inserted << " new records\n";
474
475 // Step 2: WebSocket broadcast — full DB refresh for all clients
476 wsServer_.broadcastAll();
477
478 // Step 3: Shared memory ring for JSON file writer
479 if (inserted > 0)
480 {
481 auto payload = asn_tlv::encodeSerRecordsToTlv(newRecords);
482 if (!payload.empty())
483 shmRing_.write(payload.data(), payload.size());
484 }
485 }
486 std::cout << relay_tag_ << " Worker thread exiting\n";
487 }

587
588 // Owned per-relay resources
589 TelnetClient client_;
590 RawDataRingBuffer rawBuffer_{100};
591
592 std::unique_ptr<PipelineReceptionWorker> rxWorker_;
593 std::unique_ptr<PipelineProcessingWorker> procWorker_;
594

597public:
607 RelayPipeline(const RelayConfig& config,
608 SERDatabase& db,
609 SERWebSocketServer& wsServer,
610 SharedRingBuffer& shmRing,
611 std::atomic<bool>& app_running)
612 : config_(config)
613 , db_(db)
614 , wsServer_(wsServer)
615 , shmRing_(shmRing)
616 , app_running_(app_running)
617 {
618 }
619
621 RelayPipeline(const RelayPipeline&) = delete;
622 RelayPipeline& operator=(const RelayPipeline&) = delete;
623
633 bool start()
634 {
635 if (running_)
636 return true;
637
638 ConnectionConfig conn{config_.host, config_.port, config_.timeout};
639 LoginConfig creds{config_.username, config_.password};
640
641 rxWorker_ = std::make_unique<PipelineReceptionWorker>(
642 client_, conn, creds, rawBuffer_, app_running_, config_.name);
643 rxWorker_->start();
644
645 procWorker_ = std::make_unique<PipelineProcessingWorker>(
646 rawBuffer_, db_, wsServer_, shmRing_, app_running_,
647 config_.id, config_.name);
648 procWorker_->start();
649
650 running_ = true;
651 std::cout << "[Pipeline:" << config_.name << "] Started\n";
652
653 // Queue initial SER command to start data flow
654 queueCommand("SER");
655
656 return true;
657 }
658
662 void stop()
663 {
664 if (!running_)
665 return;
666
667 if (rxWorker_)
668 rxWorker_->stop();
669 if (procWorker_)
670 procWorker_->stop();
671
672 running_ = false;
673 std::cout << "[Pipeline:" << config_.name << "] Stopped\n";
674 }
675
681 void queueCommand(const std::string& cmd)
682 {
683 if (rxWorker_)
684 rxWorker_->queueCommand(cmd);
685 }
686
694 std::string handleUserCommand(const std::string& cmd)
695 {
696 if (rxWorker_)
697 return rxWorker_->handleUserCommand(cmd);
698 return "";
699 }
700
702 bool isRunning() const { return running_; }
703
705 const RelayConfig& config() const { return config_; }
706
708 const std::string& relayId() const { return config_.id; }
709
711 const std::string& relayName() const { return config_.name; }
712
714 TelnetClient& getClient() { return client_; }
715};
