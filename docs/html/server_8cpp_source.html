318 void handleClient(tcp::socket& socket)
319 {
320 try
321 {
322 while (socket.is_open())
323 {
324 // Send prompt
325 std::string prompt = "> ";
326 boost::asio::write(socket, boost::asio::buffer(prompt));
327
328 // Receive command
329 boost::asio::streambuf buffer;
330 boost::asio::read_until(socket, buffer, '\n');
331
332 std::istream is(&buffer);
333 std::string command;
334 std::getline(is, command);
335
336 if (command.empty())
337 {
338 continue;
339 }
340
341 // Remove carriage return if present
342 if (!command.empty() && command.back() == '\r')
343 {
344 command.pop_back();
345 }
346
347 std::cout << "Client command: " << command << std::endl;
348
349 // Check for exit command
350 if (command == "exit" || command == "quit")
351 {
352 std::string bye = "Goodbye!\r\n";
353 boost::asio::write(socket, boost::asio::buffer(bye));
354 break;
355 }
356
357 // Process command and send response
358 std::string response = processCommand(command);
359 response += "\r\n";
360 boost::asio::write(socket, boost::asio::buffer(response));
361 }
362
363 socket.close();
364 std::cout << "Client disconnected.\n" << std::endl;
365 }
366 catch (const std::exception& e)
367 {
368 std::cerr << "Client handler error: " << e.what() << std::endl;
369 }
