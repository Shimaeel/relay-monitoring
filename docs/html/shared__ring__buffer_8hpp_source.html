199 bool write(const uint8_t* data, std::size_t len)
200 {
201 if (data == nullptr || len == 0U)
202 return false;
203
204 if (len > maxPayloadSize())

206
207 std::size_t readPos = read_pos_.load(std::memory_order_acquire);
208 std::size_t writePos = write_pos_.load(std::memory_order_relaxed);
209
210 readPos = normalizePos(readPos);
211 writePos = normalizePos(writePos);
212
213 if (writePos + kHeaderSize + len > capacity_)
214 {
215 writePos = 0U;
216 }
217
218 std::size_t freeSpace = freeBytes(readPos, writePos);
219 while (freeSpace < kHeaderSize + len)
220 {
221 if (!dropOldest(readPos, writePos))
222 {
223 return false;
224 }
225 overwrite_flag_.store(true, std::memory_order_relaxed);
226 freeSpace = freeBytes(readPos, writePos);

278 bool waitRead(std::vector<uint8_t>& out, std::atomic<bool>& stop_flag)
279 {

524 value |= static_cast<uint32_t>(buffer_[pos + 1U]) << 8U;
