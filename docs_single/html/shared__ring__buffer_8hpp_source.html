161 ReaderId registerReader()

227 bool write(const uint8_t* data, std::size_t len)
228 {
229 if (data == nullptr || len == 0U)
230 return false;
231
232 if (len > maxPayloadSize())
233 return false;
234
235 boost::interprocess::scoped_lock<boost::interprocess::interprocess_mutex> lock(header_->mutex);
236 std::size_t writePos = normalizePos(header_->write_pos);
237
238 // Wrap to start if message won't fit at end of buffer
239 if (writePos + kHeaderSize + len > header_->capacity)
240 {
241 writePos = 0U;
242 }
243
244 // Calculate free space based on slowest active reader
245 std::size_t slowReadPos = effectiveReadPos(writePos);
246 std::size_t free = freeBytes(slowReadPos, writePos);
247
248 while (free < kHeaderSize + len)
249 {
250 if (!dropOldestForSlowest(writePos))
251 {
252 return false;
253 }
254 header_->overwrite_flag = true;
255 ++header_->drop_count;
256
257 slowReadPos = effectiveReadPos(writePos);
258 free = freeBytes(slowReadPos, writePos);
259 }
260
261 writeLength(writePos, static_cast<uint32_t>(len));
262 std::memcpy(&buffer_[writePos + kHeaderSize], data, len);
263
264 std::size_t newWrite = writePos + kHeaderSize + len;
265 if (newWrite >= header_->capacity)
266 {
267 newWrite = 0U;
268 }
269
270 header_->write_pos = newWrite;
271 ++header_->total_writes;
272
273 // notify_all(): wake ALL waiting consumers so every reader sees the message.
274 header_->not_empty_cv.notify_all();
275 return true;
276 }

293 bool waitRead(ReaderId reader_id, std::vector<uint8_t>& out, std::atomic<bool>& stop_flag)
294 {
295 if (reader_id >= kMaxReaders)
296 return false;
297
298 boost::interprocess::scoped_lock<boost::interprocess::interprocess_mutex> lock(header_->mutex);
299
300 ReaderSlot& reader = header_->readers[reader_id];
301 if (!reader.active)
302 return false;
303
304 header_->not_empty_cv.wait(lock, [&]() {
305 return stop_flag.load(std::memory_order_relaxed)
306 || reader.read_pos != header_->write_pos;
307 });
308
309 if (stop_flag.load(std::memory_order_relaxed))
310 return false;
311
312 std::size_t readPos = normalizePos(reader.read_pos);
313 std::size_t writePos = normalizePos(header_->write_pos);
314
315 if (readPos == writePos)
316 return false;
317
318 if (readPos + kHeaderSize > header_->capacity)
319 {
320 readPos = 0U;
321 }
322
323 uint32_t len = readLength(readPos);
324 if (len == 0U || len > maxPayloadSize())
325 {
326 reader.read_pos = writePos;
327 return false;
328 }
329
330 if (readPos + kHeaderSize + len > header_->capacity)
331 {
332 readPos = 0U;
333 if (readPos == writePos)
334 {
335 reader.read_pos = readPos;
336 return false;
337 }
338 len = readLength(readPos);
339 if (len == 0U || len > maxPayloadSize())
340 {
341 reader.read_pos = writePos;
342 return false;
343 }
344 }
345
346 out.assign(buffer_ + readPos + kHeaderSize,
347 buffer_ + readPos + kHeaderSize + len);
348
349 std::size_t newRead = readPos + kHeaderSize + len;
350 if (newRead >= header_->capacity)
351 {
352 newRead = 0U;
353 }
354
355 reader.read_pos = newRead;
356 ++reader.total_reads;
357 return true;
358 }

584 std::size_t effectiveReadPos(std::size_t writePos) const
585 {
586 std::size_t slowest = writePos;
587 std::size_t maxUsed = 0U;
588 bool anyActive = false;
589
590 for (uint32_t i = 0U; i < kMaxReaders; ++i)
591 {
592 if (!header_->readers[i].active)
593 continue;
594 anyActive = true;
595 std::size_t rp = normalizePos(header_->readers[i].read_pos);
596 std::size_t used = usedBytes(rp, writePos);
597 if (used > maxUsed)
598 {
599 maxUsed = used;
600 slowest = rp;
601 }
602 }
603
604 if (!anyActive)
605 return writePos;
606
607 return slowest;
608 }

617 uint32_t slowestReaderIndex(std::size_t writePos) const
618 {
619 uint32_t slowest = kMaxReaders;
620 std::size_t maxUsed = 0U;
621
622 for (uint32_t i = 0U; i < kMaxReaders; ++i)
623 {
624 if (!header_->readers[i].active)
625 continue;
626 std::size_t rp = normalizePos(header_->readers[i].read_pos);
627 std::size_t used = usedBytes(rp, writePos);
628 if (slowest == kMaxReaders || used > maxUsed)
629 {
630 maxUsed = used;
631 slowest = i;
632 }
633 }

636
647 bool dropOldestForSlowest(std::size_t writePos)
648 {
649 uint32_t si = slowestReaderIndex(writePos);
650 if (si >= kMaxReaders)
651 return false; // No active readers
652
653 std::size_t origPos = normalizePos(header_->readers[si].read_pos);
654 if (origPos == writePos)
655 return false; // Buffer empty for this reader
656
657 std::size_t readPos = origPos;
658 if (readPos + kHeaderSize > header_->capacity)
659 {
660 readPos = 0U;
661 }
662 if (readPos == writePos)
663 return false;
664
665 uint32_t len = readLength(readPos);
666 if (len == 0U || len > maxPayloadSize())
667 {
668 // Corrupt entry — skip all readers at this position to writePos
669 for (uint32_t i = 0U; i < kMaxReaders; ++i)
670 {
671 if (header_->readers[i].active &&
672 normalizePos(header_->readers[i].read_pos) == origPos)
673 {
674 header_->readers[i].read_pos = writePos;
675 }
676 }
677 return true;
678 }
679
680 std::size_t skipTo = readPos + kHeaderSize + len;
681 if (skipTo >= header_->capacity)
682 {
683 skipTo = 0U;
684 }
685
686 // Advance ALL readers that are at the same oldest position
687 for (uint32_t i = 0U; i < kMaxReaders; ++i)
688 {
689 if (header_->readers[i].active)
690 {
691 std::size_t rp = normalizePos(header_->readers[i].read_pos);
692 if (rp == origPos)
693 header_->readers[i].read_pos = skipTo;
694 }
695 }
696 return true;
697 }

708 uint32_t readLength(std::size_t pos) const
709 {
710 uint32_t value = 0U;
711 value |= static_cast<uint32_t>(buffer_[pos]);
712 value |= static_cast<uint32_t>(buffer_[pos + 1U]) << 8U;
713 value |= static_cast<uint32_t>(buffer_[pos + 2U]) << 16U;
714 value |= static_cast<uint32_t>(buffer_[pos + 3U]) << 24U;
715 return value;
716 }
717
728 void writeLength(std::size_t pos, uint32_t len)
729 {
730 buffer_[pos] = static_cast<uint8_t>(len & 0xFFU);
731 buffer_[pos + 1U] = static_cast<uint8_t>((len >> 8U) & 0xFFU);
732 buffer_[pos + 2U] = static_cast<uint8_t>((len >> 16U) & 0xFFU);
733 buffer_[pos + 3U] = static_cast<uint8_t>((len >> 24U) & 0xFFU);
734 }

741inline SharedRingBuffer::SharedRingBuffer(const std::string& name,
742 std::size_t capacity_bytes,
743 OpenMode mode,
744 bool remove_on_destroy)
745 : name_(name), remove_on_destroy_(remove_on_destroy)
746{
747 if (name_.empty())
748 throw std::invalid_argument("SharedRingBuffer name is empty");
749
750 const std::size_t total_size = sizeof(SharedHeader) + capacity_bytes;
751
752 if (mode == OpenMode::CreateOnly)
753 {
754 boost::interprocess::shared_memory_object::remove(name_.c_str());
755 shm_ = boost::interprocess::shared_memory_object(
756 boost::interprocess::create_only, name_.c_str(), boost::interprocess::read_write);
757 shm_.truncate(static_cast<boost::interprocess::offset_t>(total_size));
758 }
759 else if (mode == OpenMode::OpenOnly)
760 {
761 shm_ = boost::interprocess::shared_memory_object(
762 boost::interprocess::open_only, name_.c_str(), boost::interprocess::read_write);
763 }
764 else
765 {
766 shm_ = boost::interprocess::shared_memory_object(
767 boost::interprocess::open_or_create, name_.c_str(), boost::interprocess::read_write);
768 shm_.truncate(static_cast<boost::interprocess::offset_t>(total_size));
769 }
770
771 region_ = boost::interprocess::mapped_region(shm_, boost::interprocess::read_write);
772 if (region_.get_size() < sizeof(SharedHeader))
773 throw std::runtime_error("SharedRingBuffer region too small");
774
775 header_ = static_cast<SharedHeader*>(region_.get_address());
776 buffer_ = reinterpret_cast<uint8_t*>(header_ + 1U);
777
778 const bool needs_init = (header_->magic != kMagic);
779 if (needs_init)
780 {
781 if (capacity_bytes <= kHeaderSize)
782 throw std::invalid_argument("SharedRingBuffer capacity too small");
783
784 header_ = new (region_.get_address()) SharedHeader();
785 header_->magic = kMagic;
786 header_->capacity = capacity_bytes;
787 header_->write_pos = 0U;
788 header_->overwrite_flag = false;
789 header_->drop_count = 0U;
790 header_->total_writes = 0U;
791
792 for (uint32_t i = 0U; i < kMaxReaders; ++i)
793 {
794 header_->readers[i].active = false;
795 header_->readers[i].read_pos = 0U;
796 header_->readers[i].total_reads = 0U;
797 }
798
799 buffer_ = reinterpret_cast<uint8_t*>(header_ + 1U);
800 std::memset(buffer_, 0, capacity_bytes);
801 }
802 else if (capacity_bytes != 0U && header_->capacity != capacity_bytes)
803 {
804 throw std::runtime_error("SharedRingBuffer capacity mismatch");
805 }
806}
